78void SplashScreen::UpdateLoading(float deltaTime) {
79 if (m_LoadingComplete || m_LoadingStages.empty()) {
80 return;
81 }
82
83 m_LoadingTime += deltaTime;
84
85 // Calculate progress through current stage
86 float currentStageDuration = m_LoadingStages[m_CurrentStageIndex].duration;
87 float stageProgress = std::min(m_LoadingTime / currentStageDuration, 1.0f);
88
89 // Calculate overall progress
90 float totalDuration = 0.0f;
91 for (const auto& stage : m_LoadingStages) {
92 totalDuration += stage.duration;
93 }
94
95 float elapsedTime = 0.0f;
96 for (size_t i = 0; i < m_CurrentStageIndex; i++) {
97 elapsedTime += m_LoadingStages[i].duration;
98 }
99 elapsedTime += m_LoadingTime;
100
101 m_LoadingProgress = std::min(elapsedTime / totalDuration, 1.0f);
102
103 // Move to next stage if current stage is complete
104 if (m_LoadingTime >= currentStageDuration) {
105 m_LoadingTime = 0.0f;
106 m_CurrentStageIndex++;
107
108 if (m_CurrentStageIndex >= m_LoadingStages.size()) {
109 m_LoadingComplete = true;
110 m_CurrentStage = "Ready!";
111 m_LoadingProgress = 1.0f;
112 } else {
113 m_CurrentStage = m_LoadingStages[m_CurrentStageIndex].name;
114 }
115 }
