58// FileSystemWatcher callback wrapper
59static void OnFileSystemEvent(const std::string& watchedPath, const FileSystemEvent& fsEvent) {
60 // Check if this is from a read-only mount
61 bool isReadOnly = IsReadOnlyPath(fsEvent.path);
62
63 // Create FileEvent
64 FileEvent event;
65 event.type = ConvertEventType(fsEvent.type);
66 event.path = fsEvent.path;
67 event.oldPath = fsEvent.oldPath;
68 event.isReadOnly = isReadOnly;
69
70 // Normalize path
71 event.path = LxPath::Normalize(event.path);
72 if (!event.oldPath.empty()) {
73 event.oldPath = LxPath::Normalize(event.oldPath);
74 }
75
76 // Queue event for debouncing
77 QueuedEvent queuedEvent;
78 queuedEvent.event = event;
79 queuedEvent.timestamp = std::chrono::steady_clock::now();
80 queuedEvent.watchedPath = watchedPath;
81
82 {
83 std::lock_guard<std::mutex> lock(s_EventQueueMutex);
84 s_EventQueue.push(queuedEvent);
85 }
86}

88// Watch a directory for changes
89bool Watch(const std::string& absRoot, FileEventCallback callback) {
90 if (absRoot.empty()) {
91 Log::Error("FSWatch: Cannot watch empty path");
92 return false;
93 }
94
95 std::string normalizedPath = LxPath::Normalize(absRoot);
96
97 // Check if already watching
98 {
99 std::lock_guard<std::mutex> lock(s_WatchedDirsMutex);
100 if (s_WatchedDirs.find(normalizedPath) != s_WatchedDirs.end()) {
101 Log::Warn("FSWatch: Already watching: " + normalizedPath);
102 return true;
103 }
104 }
105
106 // Check if path is read-only
107 bool isReadOnly = IsReadOnlyPath(normalizedPath);
108
109 // Check if this is Packages mount and watching is disabled
110 auto virtualPath = VFS::ToVirtualPath(normalizedPath);
111 if (virtualPath.has_value()) {
112 size_t colonPos = virtualPath->find(':');
113 if (colonPos != std::string::npos) {
114 std::string mountName = virtualPath->substr(0, colonPos);
115 if (mountName == "Packages" && !s_WatchPackages) {
116 Log::Info("FSWatch: Skipping Packages mount (watching disabled)");
117 return false;
118 }
119 }
120 }
121
122 // Create watcher
123 auto watcher = std::make_unique<FileSystemWatcher>();
124 watcher->SetEventCallback([normalizedPath](const FileSystemEvent& fsEvent) {
125 OnFileSystemEvent(normalizedPath, fsEvent);
126 });
127
128 if (!watcher->WatchDirectory(normalizedPath)) {
129 Log::Error("FSWatch: Failed to start watching: " + normalizedPath);
130 return false;
131 }
132
133 // Store watcher
134 {
135 std::lock_guard<std::mutex> lock(s_WatchedDirsMutex);
136 WatchedDirectory& watched = s_WatchedDirs[normalizedPath];
137 watched.path = normalizedPath;
138 watched.watcher = std::move(watcher);
139 watched.callback = callback;
140 watched.isReadOnly = isReadOnly;
141 }
142
143 Log::Info("FSWatch: Started watching: " + normalizedPath + (isReadOnly ? " (read-only)" : ""));
144 return true;
145}

170// Pump events (call from main thread)
171void Pump() {
172 auto now = std::chrono::steady_clock::now();
173 auto debounceDuration = std::chrono::milliseconds(s_DebounceIntervalMs);
174
175 // Process events that are older than debounce interval
176 std::vector<QueuedEvent> readyEvents;
177
178 {
179 std::lock_guard<std::mutex> lock(s_EventQueueMutex);
180
181 while (!s_EventQueue.empty()) {
182 QueuedEvent& queued = s_EventQueue.front();
183
184 // Check if event is old enough to process
185 auto age = now - queued.timestamp;
186 if (age >= debounceDuration) {
187 readyEvents.push_back(queued);
188 s_EventQueue.pop();
189 } else {
190 // Events are in order, so if this one isn't ready, none after it are
191 break;
192 }
193 }
194 }
195
196 // Coalesce events (remove duplicates)
197 // Group by path and keep only the latest event for each path
198 std::unordered_map<std::string, std::pair<FileEvent, std::chrono::steady_clock::time_point>> coalescedEvents;
199
200 for (const auto& queued : readyEvents) {
201 const FileEvent& event = queued.event;
202 std::string key = event.path;
203
204 // For rename events, also consider oldPath
205 if (event.type == FileEventType::Renamed && !event.oldPath.empty()) {
206 key = event.oldPath + " -> " + event.path;
207 }
208
209 // Keep the latest event for each path
210 auto it = coalescedEvents.find(key);
211 if (it == coalescedEvents.end() || queued.timestamp > it->second.second) {
212 coalescedEvents[key] = {event, queued.timestamp};
213 }
214 }
215
216 // Call callbacks for coalesced events
217 {
218 std::lock_guard<std::mutex> lock(s_WatchedDirsMutex);
219 for (const auto& [key, eventPair] : coalescedEvents) {
220 const FileEvent& event = eventPair.first;
221 // Find which watcher this event belongs to
222 for (auto& [watchedPath, watched] : s_WatchedDirs) {
223 if (LxPath::IsSubPath(event.path, watchedPath) ||
224 (!event.oldPath.empty() && LxPath::IsSubPath(event.oldPath, watchedPath))) {
225 if (watched.callback) {
226 watched.callback(event);
227 }
228 break;
229 }
230 }
231 }
232 }
233
234 // Update all watchers
235 {
236 std::lock_guard<std::mutex> lock(s_WatchedDirsMutex);
237 for (auto& [path, watched] : s_WatchedDirs) {
238 watched.watcher->Update();
239 }
240 }
241}
