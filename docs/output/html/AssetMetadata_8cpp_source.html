123bool AssetMetadata::DeserializeFromJSON(const std::string& json, LxAssetMeta& meta) {
124 // Simple JSON parser (for now - in production, use a proper JSON library)
125 // This is a basic implementation that handles the expected format
126
127 try {
128 // Find each field in the JSON
129 auto findField = [&json](const std::string& fieldName) -> std::string {
130 std::string searchStr = "\"" + fieldName + "\":";
131 size_t pos = json.find(searchStr);
132 if (pos == std::string::npos) {
133 return "";
134 }
135
136 pos += searchStr.length();
137 while (pos < json.length() && (json[pos] == ' ' || json[pos] == '\t')) {
138 pos++;
139 }
140
141 if (pos >= json.length()) {
142 return "";
143 }
144
145 if (json[pos] == '"') {
146 // String value
147 pos++;
148 size_t endPos = json.find('"', pos);
149 if (endPos == std::string::npos) {
150 return "";
151 }
152 return json.substr(pos, endPos - pos);
153 } else {
154 // Number value
155 size_t endPos = pos;
156 while (endPos < json.length() &&
157 (std::isdigit(json[endPos]) || json[endPos] == '.' || json[endPos] == '-')) {
158 endPos++;
159 }
160 return json.substr(pos, endPos - pos);
161 }
162 };
163
164 meta.name = findField("Name");
165 meta.path = findField("Path");
166 meta.relativePath = findField("RelativePath");
167 meta.type = StringToAssetType(findField("Type"));
168 meta.importer = findField("Importer");
169 meta.sourceFile = findField("SourceFile");
170 meta.guid = findField("GUID");
171
172 // Parse timestamps
173 std::string importTimeStr = findField("ImportTime");
174 std::string modifyTimeStr = findField("ModifyTime");
175 if (!importTimeStr.empty()) {
176 meta.importTime = std::stoull(importTimeStr);
177 }
178 if (!modifyTimeStr.empty()) {
179 meta.modifyTime = std::stoull(modifyTimeStr);
180 }
181
182 return true;
183
184 } catch (const std::exception& e) {
185 Log::Error("AssetMetadata: JSON parsing error: " + std::string(e.what()));
186 return false;
