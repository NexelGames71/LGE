15// Mount a path in the virtual filesystem
16void MountPath(const Mount& mount) {
17 if (mount.name.empty()) {
18 Log::Error("VFS: Cannot mount path with empty name");
19 return;
20 }
21
22 if (mount.root.empty()) {
23 Log::Error("VFS: Cannot mount path with empty root: " + mount.name);
24 return;
25 }
26
27 // Normalize the root path
28 std::string normalizedRoot = LxPath::Normalize(mount.root);
29
30 // Check if mount with this name already exists
31 auto it = std::find_if(s_Mounts.begin(), s_Mounts.end(),
32 [&mount](const Mount& m) { return m.name == mount.name; });
33
34 if (it != s_Mounts.end()) {
35 // Update existing mount
36 it->root = normalizedRoot;
37 it->readOnly = mount.readOnly;
38 it->priority = mount.priority;
39 Log::Info("VFS: Updated mount: " + mount.name + " -> " + normalizedRoot);
40 } else {
41 // Add new mount
42 Mount newMount = mount;
43 newMount.root = normalizedRoot;
44 s_Mounts.push_back(newMount);
45 Log::Info("VFS: Mounted: " + mount.name + " -> " + normalizedRoot +
46 (mount.readOnly ? " (read-only)" : " (read-write)"));
47 }
48
49 // Sort mounts by priority (lower priority = higher priority in resolution)
50 std::sort(s_Mounts.begin(), s_Mounts.end(),
51 [](const Mount& a, const Mount& b) { return a.priority < b.priority; });
52}

84// Resolve a virtual path to an absolute path
85std::optional<std::string> ResolveAbsolute(std::string_view virtualPath) {
86 if (virtualPath.empty()) {
87 return std::nullopt;
88 }
89
90 std::string vPath(virtualPath);
91
92 // Find the mount separator ":"
93 size_t colonPos = vPath.find(':');
94 if (colonPos == std::string::npos) {
95 // No mount separator, treat as relative path (try Assets first)
96 auto assetsMount = GetMount("Assets");
97 if (assetsMount.has_value()) {
98 return LxPath::Join(assetsMount->root, vPath);
99 }
100 return std::nullopt;
101 }
102
103 // Extract mount name
104 std::string mountName = vPath.substr(0, colonPos);
105
106 // Extract path after ":"
107 std::string_view pathPart = vPath.substr(colonPos + 1);
108
109 // Remove leading slash if present
110 if (!pathPart.empty() && pathPart[0] == '/') {
111 pathPart = pathPart.substr(1);
112 }
113
114 // Find the mount
115 auto mount = GetMount(mountName);
116 if (!mount.has_value()) {
117 Log::Warn("VFS: Mount not found: " + mountName);
118 return std::nullopt;
119 }
120
121 // Join mount root with path
122 std::string absolutePath = LxPath::Join(mount->root, std::string(pathPart));
123
124 return absolutePath;
125}

129 if (absolutePath.empty()) {
