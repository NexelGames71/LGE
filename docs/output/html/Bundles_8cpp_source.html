97// Helper: Read manifest from file
98static bool ReadManifest(const std::string& bundleId, BundleManifest& manifest) {
99 std::string manifestPath = GetManifestPath(bundleId);
100
101 if (!LxFile::Exists(manifestPath)) {
102 return false;
103 }
104
105 std::string jsonStr;
106 IOError err = LxFile::ReadAll(manifestPath, jsonStr);
107 if (err != IOError::None) {
108 return false;
109 }
110
111 // Simple JSON parsing (can be enhanced with proper JSON library)
112 // Extract bundleId
113 size_t pos = jsonStr.find("\"bundleId\"");
114 if (pos != std::string::npos) {
115 pos = jsonStr.find(':', pos);
116 if (pos != std::string::npos) {
117 size_t start = jsonStr.find('"', pos) + 1;
118 size_t end = jsonStr.find('"', start);
119 manifest.bundleId = jsonStr.substr(start, end - start);
120 }
121 }
122
123 // Extract name
124 pos = jsonStr.find("\"name\"");
125 if (pos != std::string::npos) {
126 pos = jsonStr.find(':', pos);
127 if (pos != std::string::npos) {
128 size_t start = jsonStr.find('"', pos) + 1;
129 size_t end = jsonStr.find('"', start);
130 manifest.name = jsonStr.substr(start, end - start);
131 }
132 }
133
134 // Extract version
135 pos = jsonStr.find("\"version\"");
136 if (pos != std::string::npos) {
137 pos = jsonStr.find(':', pos);
138 if (pos != std::string::npos) {
139 size_t start = jsonStr.find('"', pos) + 1;
140 size_t end = jsonStr.find('"', start);
141 manifest.version = jsonStr.substr(start, end - start);
142 }
143 }
144
145 // Extract assetGuids array
146 pos = jsonStr.find("\"assetGuids\"");
147 if (pos != std::string::npos) {
148 size_t arrayStart = jsonStr.find('[', pos);
149 size_t arrayEnd = jsonStr.find(']', arrayStart);
150 if (arrayStart != std::string::npos && arrayEnd != std::string::npos) {
151 std::string arrayStr = jsonStr.substr(arrayStart + 1, arrayEnd - arrayStart - 1);
152 manifest.assetGuids.clear();
153
154 size_t elemStart = 0;
155 while (elemStart < arrayStr.length()) {
156 size_t quoteStart = arrayStr.find('"', elemStart);
157 if (quoteStart == std::string::npos) break;
158 size_t quoteEnd = arrayStr.find('"', quoteStart + 1);
159 if (quoteEnd == std::string::npos) break;
160
161 std::string guid = arrayStr.substr(quoteStart + 1, quoteEnd - quoteStart - 1);
162 if (!guid.empty()) {
163 manifest.assetGuids.push_back(guid);
164 }
165
166 elemStart = quoteEnd + 1;
167 }
168 }
169 }
170
171 // Extract dependencies array
172 pos = jsonStr.find("\"dependencies\"");
173 if (pos != std::string::npos) {
174 size_t arrayStart = jsonStr.find('[', pos);
175 size_t arrayEnd = jsonStr.find(']', arrayStart);
176 if (arrayStart != std::string::npos && arrayEnd != std::string::npos) {
177 std::string arrayStr = jsonStr.substr(arrayStart + 1, arrayEnd - arrayStart - 1);
178 manifest.dependencies.clear();
179
180 size_t elemStart = 0;
181 while (elemStart < arrayStr.length()) {
182 size_t quoteStart = arrayStr.find('"', elemStart);
183 if (quoteStart == std::string::npos) break;
184 size_t quoteEnd = arrayStr.find('"', quoteStart + 1);
185 if (quoteEnd == std::string::npos) break;
186
187 std::string dep = arrayStr.substr(quoteStart + 1, quoteEnd - quoteStart - 1);
188 if (!dep.empty()) {
189 manifest.dependencies.push_back(dep);
190 }
191
192 elemStart = quoteEnd + 1;
193 }
194 }
195 }
196
197 return true;

200std::string CreateBundle(const std::string& bundleId,
201 const std::string& name,
202 const std::vector<std::string>& assetGuids,
203 const BundleOptions& options) {
204 if (bundleId.empty() || name.empty()) {
205 Log::Error("Bundles: Invalid bundle ID or name");
206 return "";
207 }
208
209 // Create bundle directory
210 std::string bundleDir = GetBundleDirectory(bundleId);
211 if (!LxFile::Exists(bundleDir)) {
212 IOError err = LxFile::CreateDirectory(bundleDir);
213 if (err != IOError::None) {
214 Log::Error("Bundles: Failed to create bundle directory: " + bundleDir);
215 return "";
216 }
217 }
218
219 BundleManifest manifest;
220 manifest.bundleId = bundleId;
221 manifest.name = name;
222 manifest.version = "1.0.0";
223 manifest.compression = options.compress ? options.compressionFormat : "none";
224
225 // Collect assets (including dependencies if requested)
226 std::unordered_set<std::string> allAssetGuids;
227
228 for (const auto& guid : assetGuids) {
229 allAssetGuids.insert(guid);
230
231 // Add dependencies if requested
232 if (options.includeDependencies) {
233 auto deps = Dependencies::GetDependencies(guid, true);
234 allAssetGuids.insert(deps.begin(), deps.end());
235 }
236 }
237
238 manifest.assetGuids.assign(allAssetGuids.begin(), allAssetGuids.end());
239
240 // Validate assets if requested
241 if (options.validateAssets) {
242 for (const auto& guid : manifest.assetGuids) {
243 ValidationResult validation = Validation::ValidateAsset(guid);
244 if (validation.status != ValidationStatus::Valid) {
245 Log::Warn("Bundles: Asset validation failed for: " + guid + " - " + validation.message);
246 }
247 }
248 }
249
250 // Calculate total size
251 manifest.totalSize = 0;
252 for (const auto& guid : manifest.assetGuids) {
253 const AssetRecord* record = AssetRegistry::FindByGUID(guid);
254 if (record) {
255 int64_t size = LxFile::GetFileSize(record->absPath);
256 if (size > 0) {
257 manifest.totalSize += size;
258 }
259 }
260 }
261
262 // Set timestamps
263 auto now = std::chrono::system_clock::now();
264 manifest.createdTime = std::chrono::duration_cast<std::chrono::seconds>(
265 now.time_since_epoch()).count();
266 manifest.modifiedTime = manifest.createdTime;
267
268 // Calculate checksum (simplified)
269 std::ostringstream checksumStream;
270 checksumStream << bundleId << manifest.totalSize << manifest.createdTime;
271 manifest.checksum = std::to_string(std::hash<std::string>{}(checksumStream.str()));
272
273 // Write manifest
274 if (options.createManifest) {
275 if (!WriteManifest(bundleId, manifest)) {
276 Log::Error("Bundles: Failed to write manifest for bundle: " + bundleId);
277 return "";
278 }
279 }
280
281 // Store in memory
282 s_Bundles[bundleId] = manifest;
283
284 Log::Info("Bundles: Created bundle: " + bundleId + " with " +
285 std::to_string(manifest.assetGuids.size()) + " assets");
286
287 return bundleId;
288}

290BundleLoadResult LoadBundle(const std::string& bundleId) {
291 BundleLoadResult result;
292 result.bundleId = bundleId;
293
294 // Check if already loaded
295 if (IsLoaded(bundleId)) {
296 result.success = true;
297 result.assetsLoaded = GetBundleAssets(bundleId).size();
298 return result;
299 }
300
301 // Load manifest
302 BundleManifest manifest;
303 if (s_Bundles.find(bundleId) != s_Bundles.end()) {
304 manifest = s_Bundles[bundleId];
305 } else {
306 if (!ReadManifest(bundleId, manifest)) {
307 result.errorMessage = "Failed to load bundle manifest: " + bundleId;
308 return result;
309 }
310 s_Bundles[bundleId] = manifest;
311 }
312
313 // Load dependencies first
314 for (const auto& depId : manifest.dependencies) {
315 BundleLoadResult depResult = LoadBundle(depId);
316 if (!depResult.success) {
317 Log::Warn("Bundles: Failed to load dependency bundle: " + depId);
318 }
319 }
320
321 // Load assets
322 std::unordered_set<std::string> loaded;
323
324 for (const auto& guid : manifest.assetGuids) {
325 const AssetRecord* record = AssetRegistry::FindByGUID(guid);
326 if (record) {
327 // Asset is already in registry, just mark as loaded
328 loaded.insert(guid);
329 result.loadedGuids.push_back(guid);
330 result.assetsLoaded++;
331 } else {
332 result.failedGuids.push_back(guid);
333 result.assetsFailed++;
334 }
335 }
336
337 // Mark bundle as loaded
338 s_LoadedBundles[bundleId] = loaded;
339
340 result.success = result.assetsFailed == 0;
341 if (result.assetsFailed > 0) {
342 result.errorMessage = "Failed to load " + std::to_string(result.assetsFailed) + " assets";
343 }
344
345 Log::Info("Bundles: Loaded bundle: " + bundleId + " (" +
346 std::to_string(result.assetsLoaded) + " assets)");
347
348 return result;
349}
