67void AssetManager::ScanAssets() {
68 std::lock_guard<std::mutex> lock(m_AssetsMutex);
69 m_Assets.clear();
70
71 try {
72 for (const auto& entry : std::filesystem::recursive_directory_iterator(m_AssetsDirectory)) {
73 if (entry.is_regular_file()) {
74 std::string filePath = entry.path().string();
75
76 // Skip metadata files
77 if (filePath.find(".lumameta") != std::string::npos) {
78 continue;
79 }
80
81 // Skip cache files
82 if (filePath.find("/Cache/") != std::string::npos ||
83 filePath.find("\\Cache\\") != std::string::npos) {
84 continue;
85 }
86
87 ProcessFile(filePath);
88 }
89 }
90
91 Log::Info("AssetManager: Scanned " + std::to_string(m_Assets.size()) + " assets");
92 } catch (const std::exception& e) {
93 Log::Error("AssetManager: Error scanning assets: " + std::string(e.what()));
94 }

97bool AssetManager::ImportFile(const std::string& sourcePath, const std::string& outputPath) {
98 if (!std::filesystem::exists(sourcePath)) {
99 Log::Error("AssetManager: Source file does not exist: " + sourcePath);
100 return false;
101 }
102
103 // Get importer for file
104 auto importer = ImporterRegistry::Get().GetImporterForFile(sourcePath);
105 if (!importer) {
106 Log::Error("AssetManager: No importer found for file: " + sourcePath);
107 return false;
108 }
109
110 // Determine output path
111 std::string finalOutputPath = outputPath;
112 if (finalOutputPath.empty()) {
113 std::filesystem::path source(sourcePath);
114 std::string fileName = source.stem().string();
115 std::string extension = source.extension().string();
116
117 // Create output path in assets directory
118 finalOutputPath = m_AssetsDirectory + "/" + fileName + extension;
119 }
120
121 // Import file
122 LxAssetMeta meta;
123 if (!importer->Import(sourcePath, finalOutputPath, meta)) {
124 Log::Error("AssetManager: Failed to import file: " + sourcePath);
125 return false;
126 }
127
128 // Set relative path
129 meta.relativePath = GetRelativePath(finalOutputPath);
130
131 // Save metadata
132 if (!AssetMetadata::SaveMetadata(finalOutputPath, meta)) {
133 Log::Error("AssetManager: Failed to save metadata for: " + finalOutputPath);
134 return false;
135 }
136
137 // Add to assets map
138 {
139 std::lock_guard<std::mutex> lock(m_AssetsMutex);
140 m_Assets[finalOutputPath] = meta;
141 }
142
143 // Generate thumbnail for imported asset
144 if (m_ThumbnailGenerator) {
145 m_ThumbnailGenerator->GenerateThumbnail(finalOutputPath, meta);
146 }
147
148 Log::Info("AssetManager: Imported file: " + sourcePath + " -> " + finalOutputPath);
149 return true;

194std::vector<LxAssetMeta*> AssetManager::GetAssetsInDirectory(const std::string& directory) {
195 std::vector<LxAssetMeta*> assets;
196
197 std::lock_guard<std::mutex> lock(m_AssetsMutex);
198
199 // Normalize the input directory path
200 std::filesystem::path dirPath(directory);
201 if (!dirPath.is_absolute()) {
202 dirPath = std::filesystem::path(m_AssetsDirectory) / dirPath;
203 }
204 if (!dirPath.is_absolute()) {
205 dirPath = std::filesystem::absolute(dirPath);
206 }
207 std::string normalizedDir = dirPath.string();
208 std::replace(normalizedDir.begin(), normalizedDir.end(), '\\', '/');
209
210 // Also normalize assets directory for comparison
211 std::string normalizedAssetsDir = m_AssetsDirectory;
212 std::replace(normalizedAssetsDir.begin(), normalizedAssetsDir.end(), '\\', '/');
213
214 // If directory is the assets directory, match assets directly in it
215 bool isRootDir = (normalizedDir == normalizedAssetsDir);
216
217 for (auto& pair : m_Assets) {
218 std::filesystem::path assetPath(pair.first);
219 std::string assetDir = assetPath.parent_path().string();
220
221 // Normalize asset directory path
222 std::string normalizedAssetDir = assetDir;
223 std::replace(normalizedAssetDir.begin(), normalizedAssetDir.end(), '\\', '/');
224
225 // Check if asset is in the requested directory
226 bool matches = false;
227 if (isRootDir) {
228 // For root directory, check if asset is directly in assets directory
229 matches = (normalizedAssetDir == normalizedAssetsDir);
230 } else {
231 // For subdirectories, check exact match or if it's a subdirectory
232 matches = (normalizedAssetDir == normalizedDir) ||
233 (normalizedAssetDir.find(normalizedDir + "/") == 0);
234 }
235
236 if (matches) {
237 assets.push_back(&pair.second);
238 }
239 }
240
241 return assets;

282void AssetManager::ProcessFile(const std::string& filePath) {
283 // Skip metadata and cache files
284 if (filePath.find(".lumameta") != std::string::npos ||
285 filePath.find("/Cache/") != std::string::npos ||
286 filePath.find("\\Cache\\") != std::string::npos) {
287 return;
288 }
289
290 // Check if file is in assets directory
291 std::string normalizedPath = filePath;
292 std::string normalizedAssetsDir = m_AssetsDirectory;
293 std::replace(normalizedPath.begin(), normalizedPath.end(), '\\', '/');
294 std::replace(normalizedAssetsDir.begin(), normalizedAssetsDir.end(), '\\', '/');
295
296 if (normalizedPath.find(normalizedAssetsDir) != 0) {
297 return; // File is not in assets directory
298 }
299
300 // Try to load existing metadata
301 LxAssetMeta meta;
302 if (AssetMetadata::LoadMetadata(filePath, meta)) {
303 // Metadata exists, update it
304 std::lock_guard<std::mutex> lock(m_AssetsMutex);
305 m_Assets[filePath] = meta;
306 Log::Info("AssetManager: Loaded asset metadata: " + filePath);
307 } else {
308 // No metadata, create new asset entry
309 std::filesystem::path path(filePath);
310 meta.name = path.stem().string();
311 meta.path = filePath;
312 meta.relativePath = GetRelativePath(filePath);
313 meta.type = GetAssetTypeFromExtension(path.extension().string());
314
315 auto now = std::chrono::system_clock::now();
316 meta.importTime = std::chrono::system_clock::to_time_t(now);
317 meta.modifyTime = std::chrono::system_clock::to_time_t(now);
318 meta.guid = std::to_string(std::hash<std::string>{}(filePath));
319
320 // Save metadata
321 AssetMetadata::SaveMetadata(filePath, meta);
322
323 std::lock_guard<std::mutex> lock(m_AssetsMutex);
324 m_Assets[filePath] = meta;
325
326 // Generate thumbnail for new asset
327 if (m_ThumbnailGenerator) {
328 m_ThumbnailGenerator->GenerateThumbnail(filePath, meta);
329 }
330
331 Log::Info("AssetManager: Created new asset entry: " + filePath);
332 }
