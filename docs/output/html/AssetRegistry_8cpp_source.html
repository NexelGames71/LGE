36// Process a single file and add/update it in the registry
37static void ProcessFile(const std::string& absPath, const std::string& mountName) {
38 try {
39 // Skip if not a regular file
40 if (!LxFile::IsFile(absPath)) {
41 return;
42 }
43
44 // Skip metadata files
45 if (absPath.length() > 9 && absPath.substr(absPath.length() - 9) == ".lumameta") {
46 return;
47 }
48
49 // Skip temp files
50 if (absPath.length() > 4 && absPath.substr(absPath.length() - 4) == ".tmp") {
51 return;
52 }
53
54 // Get file info
55 uint64_t lastWrite = 0;
56 if (LxFile::Exists(absPath)) {
57 // Get last write time from filesystem
58 auto ftime = std::filesystem::last_write_time(absPath);
59 auto sctp = std::chrono::time_point_cast<std::chrono::system_clock::duration>(
60 ftime - std::filesystem::file_time_type::clock::now() + std::chrono::system_clock::now());
61 lastWrite = std::chrono::duration_cast<std::chrono::seconds>(sctp.time_since_epoch()).count();
62 }
63
64 // Load metadata if available
65 std::optional<Meta> meta = MetaDB::Load(absPath);
66
67 AssetRecord record;
68
69 if (meta.has_value()) {
70 // Use metadata
71 record.guid = meta->guid;
72 record.type = meta->type;
73 record.tags = meta->tags;
74 record.dependencies = meta->dependencies;
75 record.lastWrite = meta->timestamp > 0 ? meta->timestamp : lastWrite;
76 } else {
77 // Create basic record
78 record.guid = MetaDB::GenerateGUID();
79 record.type = MetaDB::GetAssetTypeFromExtension(absPath);
80 record.lastWrite = lastWrite;
81 }
82
83 // Set path and name
84 record.absPath = absPath;
85 record.name = LxPath::GetStem(absPath);
86 record.mount = mountName;
87
88 // Add or update in registry
89 std::string oldGuid;
90 if (s_GUIDByPath.find(absPath) != s_GUIDByPath.end()) {
91 oldGuid = s_GUIDByPath[absPath];
92 if (oldGuid != record.guid) {
93 // GUID changed, remove old entry
94 s_AssetsByGUID.erase(oldGuid);
95 }
96 }
97
98 s_AssetsByGUID[record.guid] = record;
99 s_GUIDByPath[absPath] = record.guid;
100
101 } catch (const std::exception& e) {
102 Log::Error("AssetRegistry: Failed to process file " + absPath + ": " + e.what());
103 }
104}

106// Scan a directory recursively
107static void ScanDirectory(const std::string& absDirPath, const std::string& mountName,
108 size_t& fileCount, size_t& processedCount, ProgressCallback progressCallback) {
109 try {
110 if (!std::filesystem::exists(absDirPath) || !std::filesystem::is_directory(absDirPath)) {
111 return;
112 }
113
114 for (const auto& entry : std::filesystem::recursive_directory_iterator(absDirPath)) {
115 if (entry.is_regular_file()) {
116 fileCount++;
117
118 std::string filePath = entry.path().string();
119 std::replace(filePath.begin(), filePath.end(), '\\', '/');
120
121 ProcessFile(filePath, mountName);
122 processedCount++;
123
124 // Call progress callback every 100 files
125 if (progressCallback && processedCount % 100 == 0) {
126 progressCallback(processedCount, fileCount, filePath);
127 }
128 }
129 }
130 } catch (const std::exception& e) {
131 Log::Error("AssetRegistry: Failed to scan directory " + absDirPath + ": " + e.what());
132 }
133}

135// Perform a full scan of all assets
136void FullScan(ProgressCallback progressCallback) {
137 s_AssetsByGUID.clear();
138 s_GUIDByPath.clear();
139 s_ReverseDeps.clear();
140
141 Log::Info("AssetRegistry: Starting full scan...");
142
143 // Get all mounts
144 auto mounts = VFS::ListMounts();
145
146 size_t totalFiles = 0;
147 size_t processedFiles = 0;
148
149 // First pass: count files
150 for (const auto& mount : mounts) {
151 if (std::filesystem::exists(mount.root) && std::filesystem::is_directory(mount.root)) {
152 for (const auto& entry : std::filesystem::recursive_directory_iterator(mount.root)) {
153 if (entry.is_regular_file()) {
154 totalFiles++;
155 }
156 }
157 }
158 }
159
160 // Second pass: process files
161 for (const auto& mount : mounts) {
162 if (std::filesystem::exists(mount.root) && std::filesystem::is_directory(mount.root)) {
163 ScanDirectory(mount.root, mount.name, totalFiles, processedFiles, progressCallback);
164 }
165 }
166
167 // Build reverse dependency map
168 BuildReverseDependencyMap();
169
170 s_IsInitialized = true;
171
172 Log::Info("AssetRegistry: Full scan complete. Found " + std::to_string(s_AssetsByGUID.size()) + " assets.");
173}

198// Remove an asset from the registry
199void Remove(const std::string& absPath) {
200 if (absPath.empty()) {
201 return;
202 }
203
204 auto it = s_GUIDByPath.find(absPath);
205 if (it != s_GUIDByPath.end()) {
206 std::string guid = it->second;
207
208 // Remove from GUID map
209 s_AssetsByGUID.erase(guid);
210
211 // Remove from path map
212 s_GUIDByPath.erase(it);

214 // Remove from reverse dependencies
215 s_ReverseDeps.erase(guid);
216
217 // Remove from other assets' reverse dependencies
218 for (auto& [depGuid, dependents] : s_ReverseDeps) {
219 dependents.erase(
220 std::remove(dependents.begin(), dependents.end(), guid),
221 dependents.end()
222 );
223 }

259// Search assets by name and tags (fuzzy search)
260std::vector<AssetRecord> SearchFuzzy(std::string_view text, size_t maxResults) {
261 std::vector<AssetRecord> results;
262
263 if (text.empty()) {
264 return results;
265 }
266
267 for (const auto& [guid, record] : s_AssetsByGUID) {
268 bool matches = false;
269
270 // Check name
271 if (ContainsIgnoreCase(record.name, text)) {
272 matches = true;
273 }
274
275 // Check tags
276 if (!matches) {
277 for (const auto& tag : record.tags) {
278 if (ContainsIgnoreCase(tag, text)) {
279 matches = true;
280 break;
281 }
282 }
283 }
284
285 // Check type
286 if (!matches) {
287 if (ContainsIgnoreCase(record.type, text)) {
288 matches = true;
289 }
290 }
291
292 if (matches) {
293 results.push_back(record);
294 if (results.size() >= maxResults) {
295 break;
296 }
297 }
298 }
299
300 return results;
301}

358// Build reverse dependency map
359void BuildReverseDependencyMap() {
360 s_ReverseDeps.clear();
361
362 // For each asset, check its dependencies
363 for (const auto& [guid, record] : s_AssetsByGUID) {
364 for (const auto& depVirtualPath : record.dependencies) {
365 // Resolve virtual path to absolute
366 auto depAbsPath = VFS::ResolveAbsolute(depVirtualPath);
367 if (depAbsPath.has_value()) {
368 // Find asset by path
369 const AssetRecord* depRecord = FindByPath(*depAbsPath);
370 if (depRecord) {
371 // Add this asset to the dependent's reverse dependency list
372 s_ReverseDeps[depRecord->guid].push_back(guid);
373 }
374 }
375 }
376 }
377
378 Log::Info("AssetRegistry: Built reverse dependency map");
379}
