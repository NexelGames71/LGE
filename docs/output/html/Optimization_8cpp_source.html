28// Helper: Calculate quality loss (simplified)
29static float CalculateQualityLoss(size_t originalSize, size_t optimizedSize, OptimizationQuality quality) {
30 if (quality == OptimizationQuality::Lossless) {
31 return 0.0f; // No quality loss
32 }
33
34 float ratio = static_cast<float>(optimizedSize) / originalSize;
35
36 // Estimate quality loss based on compression ratio
37 if (ratio < 0.5f) {
38 return 0.3f; // High compression, some quality loss
39 } else if (ratio < 0.8f) {
40 return 0.1f; // Medium compression, minimal quality loss
41 } else {
42 return 0.0f; // Low compression, no quality loss
43 }
44}

61OptimizationResult OptimizeAsset(const std::string& assetGuid,
62 const OptimizationOptions& options) {
63 OptimizationResult result;
64 result.success = false;
65 result.assetGuid = assetGuid;
66
67 auto startTime = GetCurrentTime();
68
69 // Find asset
70 const AssetRecord* record = AssetRegistry::FindByGUID(assetGuid);
71 if (!record) {
72 result.errorMessage = "Asset not found: " + assetGuid;
73 return result;
74 }
75
76 result.originalPath = record->absPath;
77
78 // Check if asset exists
79 if (!LxFile::Exists(record->absPath)) {
80 result.errorMessage = "Asset file does not exist: " + record->absPath;
81 return result;
82 }
83
84 // Get original size
85 result.originalSize = LxFile::GetFileSize(record->absPath);
86 if (result.originalSize <= 0) {
87 result.errorMessage = "Invalid file size";
88 return result;
89 }
90
91 // Determine output path
92 std::string outputPath = options.outputPath;
93 if (outputPath.empty()) {
94 if (options.overwriteOriginal) {
95 outputPath = record->absPath;
96 } else {
97 // Create optimized version in same directory
98 std::string ext = LxPath::GetExtension(record->absPath);
99 std::string stem = LxPath::GetStem(record->absPath);
100 std::string dir = LxPath::GetParent(record->absPath);
101 outputPath = LxPath::Join(dir, stem + "_optimized" + ext);
102 }
103 }
104
105 // Create backup if requested
106 if (options.createBackup && options.overwriteOriginal) {
107 std::string backupPath = record->absPath + ".backup";
108 IOError err = LxFile::Copy(record->absPath, backupPath, false);
109 if (err != IOError::None) {
110 Log::Warn("Optimization: Failed to create backup: " + backupPath);
111 }
112 }
113
114 // Optimize based on target
115 bool optimized = false;
116
117 switch (options.target) {
118 case OptimizationTarget::Size: {
119 // Optimize for size (compression)
120 CompressionOptions compOptions;
121 compOptions.format = CompressionFormat::Zip;
122 compOptions.level = GetCompressionLevel(options.quality);
123 compOptions.overwriteExisting = options.overwriteOriginal;
124
125 CompressionResult compResult = Compression::CompressFile(record->absPath, compOptions);
126 if (compResult.success) {
127 result.optimizedPath = compResult.outputPath;
128 result.optimizedSize = compResult.compressedSize;
129 result.compressionRatio = compResult.compressionRatio;
130 optimized = true;
131 } else {
132 result.errorMessage = "Compression failed: " + compResult.errorMessage;
133 }
134 break;
135 }
136 case OptimizationTarget::Quality: {
137 // For quality optimization, we'd need format-specific optimizers
138 // For now, just copy the file
139 IOError err = LxFile::Copy(record->absPath, outputPath, options.overwriteOriginal);
140 if (err == IOError::None) {
141 result.optimizedPath = outputPath;
142 result.optimizedSize = result.originalSize;
143 result.compressionRatio = 1.0f;
144 optimized = true;
145 } else {
146 result.errorMessage = "Copy failed: " + std::to_string(static_cast<int>(err));
147 }
148 break;
149 }
150 case OptimizationTarget::Speed: {
151 // For speed optimization, we'd optimize for faster loading
152 // For now, just copy the file
153 IOError err = LxFile::Copy(record->absPath, outputPath, options.overwriteOriginal);
154 if (err == IOError::None) {
155 result.optimizedPath = outputPath;
156 result.optimizedSize = result.originalSize;
157 result.compressionRatio = 1.0f;
158 optimized = true;
159 } else {
160 result.errorMessage = "Copy failed: " + std::to_string(static_cast<int>(err));
161 }
162 break;
163 }
164 case OptimizationTarget::Balanced: {
165 // Balanced optimization (compression with quality preservation)
166 CompressionOptions compOptions;
167 compOptions.format = CompressionFormat::Zip;
168 compOptions.level = CompressionLevel::Balanced;
169 compOptions.overwriteExisting = options.overwriteOriginal;
170
171 CompressionResult compResult = Compression::CompressFile(record->absPath, compOptions);
172 if (compResult.success) {
173 result.optimizedPath = compResult.outputPath;
174 result.optimizedSize = compResult.compressedSize;
175 result.compressionRatio = compResult.compressionRatio;
176 optimized = true;
177 } else {
178 result.errorMessage = "Compression failed: " + compResult.errorMessage;
179 }
180 break;
181 }
182 }
183
184 if (!optimized) {
185 return result;
186 }
187
188 // Calculate quality loss
189 result.qualityLoss = CalculateQualityLoss(result.originalSize, result.optimizedSize, options.quality);
190
191 auto endTime = GetCurrentTime();
192 result.optimizationTime = endTime - startTime;
193
194 result.success = true;
195
196 // Update statistics
197 s_Stats.totalOptimized++;
198 s_Stats.totalBytesSaved += (result.originalSize - result.optimizedSize);
199
200 // Update average compression ratio
201 if (s_Stats.totalOptimized > 0) {
202 s_Stats.averageCompressionRatio =
203 (s_Stats.averageCompressionRatio * (s_Stats.totalOptimized - 1) + result.compressionRatio) /
204 s_Stats.totalOptimized;
205
206 s_Stats.averageQualityLoss =
207 (s_Stats.averageQualityLoss * (s_Stats.totalOptimized - 1) + result.qualityLoss) /
208 s_Stats.totalOptimized;
209 }
210
211 s_Stats.totalOptimizationTime += result.optimizationTime;
212
213 // Cache result
214 s_OptimizationCache[assetGuid] = result;
215
216 Log::Info("Optimization: Optimized asset " + assetGuid +
217 " (size: " + std::to_string(result.originalSize) + " -> " +
218 std::to_string(result.optimizedSize) + " bytes, " +
219 std::to_string(result.compressionRatio * 100.0f) + "% of original)");
220
221 return result;
222}

224std::vector<OptimizationResult> OptimizeAssets(const std::vector<std::string>& assetGuids,
225 const OptimizationOptions& options) {
226 std::vector<OptimizationResult> results;
227 results.reserve(assetGuids.size());
228
229 size_t total = assetGuids.size();
230 size_t current = 0;
231
232 for (const auto& guid : assetGuids) {
233 current++;
234
235 // Call progress callback if provided
236 if (options.progressCallback) {
237 options.progressCallback(current, total);
238 }
239
240 OptimizationResult result = OptimizeAsset(guid, options);
241 results.push_back(result);
242
243 if (!result.success) {
244 s_Stats.totalFailed++;
245 }
246 }
247
248 Log::Info("Optimization: Batch optimization complete: " +
249 std::to_string(results.size()) + " assets");
250
251 return results;
252}
