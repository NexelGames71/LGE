80// Levenshtein distance calculation
81int SearchQuery::LevenshteinDistance(std::string_view a, std::string_view b) {
82 size_t lenA = a.length();
83 size_t lenB = b.length();
84
85 if (lenA == 0) return static_cast<int>(lenB);
86 if (lenB == 0) return static_cast<int>(lenA);
87
88 // Use dynamic programming
89 std::vector<std::vector<int>> dp(lenA + 1, std::vector<int>(lenB + 1));
90
91 // Initialize first row and column
92 for (size_t i = 0; i <= lenA; ++i) {
93 dp[i][0] = static_cast<int>(i);
94 }
95 for (size_t j = 0; j <= lenB; ++j) {
96 dp[0][j] = static_cast<int>(j);
97 }
98
99 // Fill the matrix
100 for (size_t i = 1; i <= lenA; ++i) {
101 for (size_t j = 1; j <= lenB; ++j) {
102 int cost = (a[i - 1] == b[j - 1]) ? 0 : 1;
103 dp[i][j] = std::min({
104 dp[i - 1][j] + 1, // deletion
105 dp[i][j - 1] + 1, // insertion
106 dp[i - 1][j - 1] + cost // substitution
107 });
108 }
109 }
110
111 return dp[lenA][lenB];
112}

114bool SearchQuery::FuzzyMatch(std::string_view text, std::string_view pattern) const {
115 if (pattern.empty()) {
116 return true;
117 }
118
119 if (text.empty()) {
120 return false;
121 }
122
123 // Normalize strings for comparison
124 std::string textStr(text);
125 std::string patternStr(pattern);
126
127 if (!m_CaseSensitive) {
128 std::transform(textStr.begin(), textStr.end(), textStr.begin(), ::tolower);
129 std::transform(patternStr.begin(), patternStr.end(), patternStr.begin(), ::tolower);
130 }
131
132 // Check for exact match first
133 if (textStr == patternStr) {
134 return true;
135 }
136
137 // Check for substring match
138 if (textStr.find(patternStr) != std::string::npos) {
139 return true;
140 }
141
142 // If fuzzy matching is enabled, check Levenshtein distance
143 if (m_Fuzzy) {
144 int distance = LevenshteinDistance(textStr, patternStr);
145 int maxLen = static_cast<int>(std::max(textStr.length(), patternStr.length()));
146
147 if (maxLen == 0) {
148 return true;
149 }
150
151 float similarity = 1.0f - (static_cast<float>(distance) / static_cast<float>(maxLen));
152 return similarity >= (1.0f - m_FuzzyThreshold);
153 }
154
155 return false;
156}

158float SearchQuery::CalculateRelevance(const AssetRecord& record, const std::string& matchReason) const {
159 float score = 0.5f; // Base score
160
161 if (m_Text.empty()) {
162 return score;
163 }
164
165 // Boost score based on match type
166 if (matchReason == "name_exact") {
167 score = 1.0f;
168 } else if (matchReason == "name_prefix") {
169 score = 0.9f;
170 } else if (matchReason == "name_contains") {
171 score = 0.8f;
172 } else if (matchReason == "name_fuzzy") {
173 // Calculate fuzzy similarity
174 std::string nameLower = record.name;
175 std::string textLower = m_Text;
176 std::transform(nameLower.begin(), nameLower.end(), nameLower.begin(), ::tolower);
177 std::transform(textLower.begin(), textLower.end(), textLower.begin(), ::tolower);
178
179 int distance = LevenshteinDistance(nameLower, textLower);
180 int maxLen = static_cast<int>(std::max(nameLower.length(), textLower.length()));
181 if (maxLen > 0) {
182 float similarity = 1.0f - (static_cast<float>(distance) / static_cast<float>(maxLen));
183 score = 0.7f * similarity;
184 }
185 } else if (matchReason == "tag") {
186 score = 0.6f;
187 } else if (matchReason == "type") {
188 score = 0.5f;
189 }
190
191 return std::clamp(score, 0.0f, 1.0f);
192}

194bool SearchQuery::Matches(const AssetRecord& record) const {
195 // Check type filter
196 if (m_Type.has_value() && record.type != m_Type.value()) {
197 return false;
198 }
199
200 // Check mount filter
201 if (m_Mount.has_value() && record.mount != m_Mount.value()) {
202 return false;
203 }
204
205 // Check path prefix filter
206 if (m_PathPrefix.has_value()) {
207 std::string normalizedPrefix = LxPath::Normalize(m_PathPrefix.value());
208 if (!LxPath::IsSubPath(record.absPath, normalizedPrefix)) {
209 return false;
210 }
211 }
212
213 // Check date range
214 if (m_MinTimestamp.has_value() && record.lastWrite < m_MinTimestamp.value()) {
215 return false;
216 }
217 if (m_MaxTimestamp.has_value() && record.lastWrite > m_MaxTimestamp.value()) {
218 return false;
219 }
220
221 // Check tags (all must be present)
222 if (!m_Tags.empty()) {
223 for (const auto& requiredTag : m_Tags) {
224 bool found = false;
225 for (const auto& tag : record.tags) {
226 if (m_CaseSensitive) {
227 if (tag == requiredTag) {
228 found = true;
229 break;
230 }
231 } else {
232 std::string tagLower = tag;
233 std::string requiredLower = requiredTag;
234 std::transform(tagLower.begin(), tagLower.end(), tagLower.begin(), ::tolower);
235 std::transform(requiredLower.begin(), requiredLower.end(), requiredLower.begin(), ::tolower);
236 if (tagLower == requiredLower) {
237 found = true;
238 break;
239 }
240 }
241 }
242 if (!found) {
243 return false;
244 }
245 }
246 }
247
248 // Check text search
249 if (!m_Text.empty()) {
250 bool textMatches = false;
251 std::string matchReason;
252
253 // Check name
254 if (FuzzyMatch(record.name, m_Text)) {
255 textMatches = true;
256 if (record.name == m_Text) {
257 matchReason = "name_exact";
258 } else if (record.name.length() >= m_Text.length() &&
259 record.name.substr(0, m_Text.length()) == m_Text) {
260 matchReason = "name_prefix";
261 } else if (record.name.find(m_Text) != std::string::npos) {
262 matchReason = "name_contains";
263 } else {
264 matchReason = "name_fuzzy";
265 }
266 }
267
268 // Check tags
269 if (!textMatches) {
270 for (const auto& tag : record.tags) {
271 if (FuzzyMatch(tag, m_Text)) {
272 textMatches = true;
273 matchReason = "tag";
274 break;
275 }
276 }
277 }
278
279 // Check type
280 if (!textMatches) {
281 if (FuzzyMatch(record.type, m_Text)) {
282 textMatches = true;
283 matchReason = "type";
284 }
285 }
286
287 if (!textMatches) {
288 return false;
289 }
290 }
291
292 return true;
293}

295std::vector<SearchResult> SearchQuery::Execute() const {
296 std::vector<SearchResult> results;
297
298 // Get all assets
299 auto allAssets = AssetRegistry::GetAllAssets();
300
301 // Filter and score
302 for (const auto& asset : allAssets) {
303 if (Matches(asset)) {
304 // Determine match reason
305 std::string matchReason = "unknown";
306
307 if (!m_Text.empty()) {
308 if (asset.name == m_Text) {
309 matchReason = "name_exact";
310 } else if (asset.name.length() >= m_Text.length() &&
311 asset.name.substr(0, m_Text.length()) == m_Text) {
312 matchReason = "name_prefix";
313 } else if (asset.name.find(m_Text) != std::string::npos) {
314 matchReason = "name_contains";
315 } else if (FuzzyMatch(asset.name, m_Text)) {
316 matchReason = "name_fuzzy";
317 } else {
318 // Check tags and type
319 bool foundInTag = false;
320 for (const auto& tag : asset.tags) {
321 if (FuzzyMatch(tag, m_Text)) {
322 matchReason = "tag";
323 foundInTag = true;
324 break;
325 }
326 }
327 if (!foundInTag && FuzzyMatch(asset.type, m_Text)) {
328 matchReason = "type";
329 }
330 }
331 }
332
333 float relevance = CalculateRelevance(asset, matchReason);
334 results.emplace_back(asset, relevance, matchReason);
335 }
336 }
337
338 // Sort by relevance (highest first)
339 std::sort(results.begin(), results.end(),
340 [](const SearchResult& a, const SearchResult& b) {
341 return a.relevanceScore > b.relevanceScore;
342 });
343
344 // Limit results
345 if (results.size() > m_MaxResults) {
346 results.resize(m_MaxResults);
347 }
348
349 return results;
350}
