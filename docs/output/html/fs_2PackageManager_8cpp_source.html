48 if (!LxFile::Exists(filePath)) {

51

66IOError ExtractPackage(const std::string& zipPath, const std::string& packageId) {
67 if (zipPath.empty() || packageId.empty()) {
68 return IOError::Invalid;
69 }
70
71 if (!LxFile::Exists(zipPath)) {
72 Log::Error("PackageMgr: ZIP file not found: " + zipPath);
73 return IOError::NotFound;
74 }
75
76 std::string packagesDir = GetPackagesDirectory();
77 if (packagesDir.empty()) {
78 Log::Error("PackageMgr: Packages directory not available");
79 return IOError::Invalid;
80 }
81
82 std::string packageDir = LxPath::Join(packagesDir, packageId);
83
84 // Create package directory directly using filesystem (bypass VFS read-only check)
85 // Packages need to be writable for installation, even though the mount is read-only
86 try {
87 if (!std::filesystem::exists(packageDir)) {
88 std::filesystem::create_directories(packageDir);
89 Log::Info("PackageMgr: Created package directory: " + packageDir);
90 }
91 } catch (const std::exception& e) {
92 Log::Error("PackageMgr: Failed to create package directory: " + packageDir + " - " + e.what());
93 return IOError::IO;
94 }
95
96 // Open ZIP file
97 int zipError = 0;
98 zip_t* zip = zip_open(zipPath.c_str(), ZIP_RDONLY, &zipError);
99 if (!zip) {
100 Log::Error("PackageMgr: Failed to open ZIP file: " + zipPath);
101 return IOError::IO;
102 }
103
104 // Get number of entries
105 zip_int64_t numEntries = zip_get_num_entries(zip, ZIP_FL_UNCHANGED);
106
107 // Extract all entries
108 for (zip_int64_t i = 0; i < numEntries; ++i) {
109 const char* name = zip_get_name(zip, i, ZIP_FL_ENC_RAW);
110 if (!name) {
111 continue;
112 }
113
114 // Skip directory entries
115 if (name[strlen(name) - 1] == '/') {
116 continue;
117 }
118
119 // Build destination path
120 std::string entryName(name);
121 // Normalize path separators
122 std::replace(entryName.begin(), entryName.end(), '\\', '/');
123
124 // Remove leading directory if present (e.g., "package-name-1.0.0/")
125 size_t firstSlash = entryName.find('/');
126 if (firstSlash != std::string::npos) {
127 entryName = entryName.substr(firstSlash + 1);
128 }
129
130 std::string destPath = LxPath::Join(packageDir, entryName);
131
132 // Create parent directory if needed (bypass VFS read-only check)
133 std::string parentDir = LxPath::GetParent(destPath);
134 if (!parentDir.empty() && !std::filesystem::exists(parentDir)) {
135 try {
136 std::filesystem::create_directories(parentDir);
137 } catch (const std::exception& e) {
138 Log::Warn("PackageMgr: Failed to create parent directory: " + parentDir + " - " + e.what());
139 }
140 }
141
142 // Open entry
143 zip_file_t* file = zip_fopen_index(zip, i, 0);
144 if (!file) {
145 Log::Warn("PackageMgr: Failed to open ZIP entry: " + std::string(name));
146 continue;
147 }
148
149 // Read entry data
150 zip_stat_t stat;
151 if (zip_stat_index(zip, i, ZIP_FL_ENC_RAW, &stat) == 0) {
152 std::vector<char> buffer(stat.size);
153 zip_int64_t read = zip_fread(file, buffer.data(), stat.size);
154
155 if (read == stat.size) {
156 // Write to destination directly using filesystem (bypass VFS read-only check)
157 try {
158 // Ensure parent directory exists
159 std::string parentDir = LxPath::GetParent(destPath);
160 if (!parentDir.empty() && !std::filesystem::exists(parentDir)) {
161 std::filesystem::create_directories(parentDir);
162 }
163
164 // Write file directly
165 std::ofstream file(destPath, std::ios::binary | std::ios::trunc);
166 if (file.is_open()) {
167 file.write(buffer.data(), buffer.size());
168 file.close();
169 } else {
170 Log::Warn("PackageMgr: Failed to open file for writing: " + destPath);
171 }
172 } catch (const std::exception& e) {
173 Log::Warn("PackageMgr: Failed to write file: " + destPath + " - " + e.what());
174 }
175 }
176 }
177
178 zip_fclose(file);
179 }
180
181 zip_close(zip);
182
183 Log::Info("PackageMgr: Extracted package to: " + packageDir);
184 return IOError::None;
185}
