70CompressionFormat GetFormatFromExtension(const std::string& filePath) {

94CompressionResult CompressFile(const std::string& filePath,
95 const CompressionOptions& options) {
96 CompressionResult result;
97
98 auto startTime = std::chrono::high_resolution_clock::now();
99
100 // Check if source exists
101 if (!LxFile::Exists(filePath)) {
102 result.errorMessage = "Source file does not exist: " + filePath;
103 return result;
104 }
105
106 // Get original size
107 result.originalSize = LxFile::GetFileSize(filePath);
108 if (result.originalSize <= 0) {
109 result.errorMessage = "Invalid file size";
110 return result;
111 }
112
113 // Determine output path
114 std::string outputPath = options.outputPath;
115 if (outputPath.empty()) {
116 std::string ext = FormatToString(options.format);
117 std::transform(ext.begin(), ext.end(), ext.begin(), ::tolower);
118 outputPath = filePath + "." + ext;
119 }
120
121 // Check if output exists
122 if (LxFile::Exists(outputPath) && !options.overwriteExisting) {
123 result.errorMessage = "Output file already exists: " + outputPath;
124 return result;
125 }
126
127 // Compress based on format
128 bool success = false;
129
130 switch (options.format) {
131 case CompressionFormat::Zip: {
132 // Use libzip for ZIP compression
133 int zipError = 0;
134 zip_t* zip = zip_open(outputPath.c_str(), ZIP_CREATE | ZIP_TRUNCATE, &zipError);
135 if (zip) {
136 std::string filename = LxPath::GetFilename(filePath);
137
138 // Read source file
139 std::string fileData;
140 IOError readErr = LxFile::ReadAll(filePath, fileData);
141 if (readErr == IOError::None) {
142 zip_source_t* source = zip_source_buffer(zip, fileData.data(), fileData.size(), 0);
143 if (source) {
144 zip_int64_t index = zip_file_add(zip, filename.c_str(), source, ZIP_FL_OVERWRITE);
145 if (index >= 0) {
146 // Set compression level
147 zip_set_file_compression(zip, index, ZIP_CM_DEFLATE, GetZipCompressionLevel(options.level));
148 success = true;
149 } else {
150 zip_source_free(source);
151 }
152 }
153 }
154 zip_close(zip);
155 }
156 break;
157 }
158 case CompressionFormat::None: {
159 // No compression - just copy
160 IOError err = LxFile::Copy(filePath, outputPath, options.overwriteExisting);
161 success = (err == IOError::None);
162 break;
163 }
164 default: {
165 // Other formats not yet implemented
166 result.errorMessage = "Compression format not yet implemented: " + FormatToString(options.format);
167 return result;
168 }
169 }
170
171 if (!success) {
172 result.errorMessage = "Compression failed";
173 return result;
174 }
175
176 // Get compressed size
177 result.compressedSize = LxFile::GetFileSize(outputPath);
178 result.compressionRatio = result.compressedSize > 0 ?
179 static_cast<float>(result.compressedSize) / result.originalSize : 0.0f;
180 result.outputPath = outputPath;
181
182 auto endTime = std::chrono::high_resolution_clock::now();
183 result.compressionTime = std::chrono::duration_cast<std::chrono::milliseconds>(
184 endTime - startTime).count();
185
186 result.success = true;
187
188 // Update statistics
189 s_Stats.totalFilesCompressed++;
190 s_Stats.totalBytesCompressed += result.originalSize;
191 s_Stats.totalBytesSaved += (result.originalSize - result.compressedSize);
192
193 // Update average compression ratio
194 if (s_Stats.totalFilesCompressed > 0) {
195 s_Stats.averageCompressionRatio =
196 (s_Stats.averageCompressionRatio * (s_Stats.totalFilesCompressed - 1) + result.compressionRatio) /
197 s_Stats.totalFilesCompressed;
198 }
199
200 s_Stats.totalCompressionTime += result.compressionTime;
201
202 Log::Info("Compression: Compressed " + filePath + " -> " + outputPath +
203 " (" + std::to_string(result.compressionRatio * 100.0f) + "% of original)");
204
205 return result;
206}
