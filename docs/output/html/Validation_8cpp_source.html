21// Helper: Calculate MD5 hash (simplified implementation)
22static std::string CalculateMD5(const std::string& filePath) {
23 std::ifstream file(filePath, std::ios::binary);
24 if (!file.is_open()) {
25 return "";
26 }
27
28 // Simple hash calculation (not real MD5, but sufficient for integrity checks)
29 std::stringstream ss;
30 std::hash<std::string> hasher;
31
32 char buffer[4096];
33 size_t totalHash = 0;
34 size_t fileSize = 0;
35
36 while (file.read(buffer, sizeof(buffer)) || file.gcount() > 0) {
37 size_t bytesRead = file.gcount();
38 fileSize += bytesRead;
39
40 // Simple hash of buffer
41 std::string bufferStr(buffer, bytesRead);
42 totalHash ^= hasher(bufferStr) + 0x9e3779b9 + (totalHash << 6) + (totalHash >> 2);
43 }
44
45 // Combine file size and hash
46 ss << std::hex << std::setfill('0') << std::setw(16) << totalHash;
47 ss << std::hex << std::setfill('0') << std::setw(16) << fileSize;
48
49 return ss.str();
50}

84ValidationResult ValidateAsset(const std::string& assetGuid) {
85 ValidationResult result;
86 result.assetGuid = assetGuid;
87 result.status = ValidationStatus::Valid;
88
89 // Check cache
90 auto cacheIt = s_ValidationCache.find(assetGuid);
91 if (cacheIt != s_ValidationCache.end()) {
92 return cacheIt->second;
93 }
94
95 // Find asset
96 const AssetRecord* record = AssetRegistry::FindByGUID(assetGuid);
97 if (!record) {
98 result.status = ValidationStatus::Missing;
99 result.message = "Asset not found in registry";
100 result.issues.push_back("Asset not found in registry");
101 s_ValidationCache[assetGuid] = result;
102 return result;
103 }
104
105 result.assetPath = record->absPath;
106
107 // Check if file exists
108 if (!LxFile::Exists(record->absPath)) {
109 result.status = ValidationStatus::Missing;
110 result.message = "Asset file does not exist: " + record->absPath;
111 result.issues.push_back("File missing");
112 s_ValidationCache[assetGuid] = result;
113 return result;
114 }
115
116 // Check if file is readable
117 if (!LxFile::IsFile(record->absPath)) {
118 result.status = ValidationStatus::Invalid;
119 result.message = "Asset path is not a file: " + record->absPath;
120 result.issues.push_back("Path is not a file");
121 s_ValidationCache[assetGuid] = result;
122 return result;
123 }
124
125 // Check metadata
126 bool metadataExists = MetaDB::Exists(record->absPath);
127 bool requiresMetadata = MetaDB::RequiresMetadata(record->absPath);
128
129 if (requiresMetadata && !metadataExists) {
130 result.status = ValidationStatus::MissingMetadata;
131 result.message = "Metadata file missing for asset: " + record->absPath;
132 result.issues.push_back("Metadata missing");
133 result.canRepair = true;
134 result.repairAction = "Create metadata file";
135 } else if (metadataExists) {
136 // Validate metadata
137 auto meta = MetaDB::Load(record->absPath);
138 if (!meta.has_value()) {
139 result.status = ValidationStatus::InvalidMetadata;
140 result.message = "Metadata file is corrupted: " + record->absPath;
141 result.issues.push_back("Metadata corrupted");
142 result.canRepair = true;
143 result.repairAction = "Repair metadata file";
144 } else {
145 // Check if GUID matches
146 if (meta->guid != record->guid) {
147 result.status = ValidationStatus::InvalidMetadata;
148 result.message = "Metadata GUID mismatch";
149 result.issues.push_back("GUID mismatch");
150 }
151 }
152 }
153
154 // Check dependencies
155 auto brokenRefs = Dependencies::ValidateAssetReferences(assetGuid);
156 if (!brokenRefs.empty()) {
157 if (result.status == ValidationStatus::Valid) {
158 result.status = ValidationStatus::BrokenDependencies;
159 }
160 result.message = "Asset has broken dependencies";
161 for (const auto& broken : brokenRefs) {
162 result.issues.push_back("Broken dependency: " + broken.reference);
163 }
164 }
165
166 // Check for circular dependencies
167 if (Dependencies::HasCircularDependency(assetGuid)) {
168 if (result.status == ValidationStatus::Valid) {
169 result.status = ValidationStatus::CircularDependency;
170 }
171 result.message = "Asset has circular dependencies";
172 result.issues.push_back("Circular dependency detected");
173 }
174
175 // If no issues found, asset is valid
176 if (result.issues.empty()) {
177 result.status = ValidationStatus::Valid;
178 result.message = "Asset is valid";
179 }
180
181 s_ValidationCache[assetGuid] = result;
182 return result;
183}

246IntegrityResult CheckIntegrity(const std::string& assetGuid) {
247 IntegrityResult result;
248 result.valid = false;
249
250 // Find asset
251 const AssetRecord* record = AssetRegistry::FindByGUID(assetGuid);
252 if (!record) {
253 result.errorMessage = "Asset not found in registry";
254 return result;
255 }
256
257 // Check if file exists
258 if (!LxFile::Exists(record->absPath)) {
259 result.errorMessage = "Asset file does not exist";
260 return result;
261 }
262
263 // Get file size
264 result.fileSize = LxFile::GetFileSize(record->absPath);
265 if (result.fileSize < 0) {
266 result.errorMessage = "Failed to get file size";
267 return result;
268 }
269
270 // Get last modified time
271 std::filesystem::path path(record->absPath);
272 if (std::filesystem::exists(path)) {
273 auto ftime = std::filesystem::last_write_time(path);
274 result.lastModified = std::chrono::duration_cast<std::chrono::seconds>(
275 ftime.time_since_epoch()).count();
276 }
277
278 // Calculate checksum
279 result.checksum = CalculateMD5(record->absPath);
280 if (result.checksum.empty()) {
281 result.errorMessage = "Failed to calculate checksum";
282 return result;
283 }
284
285 // Check metadata
286 if (MetaDB::RequiresMetadata(record->absPath)) {
287 auto meta = MetaDB::Load(record->absPath);
288 if (meta.has_value()) {
289 result.metadataMatches = true;
290 } else {
291 result.metadataMatches = false;
292 result.errorMessage = "Metadata does not match";
293 }
294 } else {
295 result.metadataMatches = true; // No metadata required
296 }
297
298 result.valid = true;
299 return result;
300}

302HealthResult CheckHealth(const std::string& assetGuid) {
303 HealthResult result;
304 result.healthy = true;
305
306 // Validate asset
307 ValidationResult validation = ValidateAsset(assetGuid);
308
309 if (validation.status != ValidationStatus::Valid) {
310 result.healthy = false;
311 result.issues.push_back(validation);
312
313 // Check if critical
314 if (validation.status == ValidationStatus::Missing ||
315 validation.status == ValidationStatus::Corrupted) {
316 result.hasCriticalIssues = true;
317 }
318 }
319
320 // Check integrity
321 IntegrityResult integrity = CheckIntegrity(assetGuid);
322 if (!integrity.valid) {
323 result.healthy = false;
324 ValidationResult integrityIssue;
325 integrityIssue.assetGuid = assetGuid;
326 integrityIssue.status = ValidationStatus::Corrupted;
327 integrityIssue.message = "Integrity check failed: " + integrity.errorMessage;
328 integrityIssue.issues.push_back(integrity.errorMessage);
329 result.issues.push_back(integrityIssue);
330 result.hasCriticalIssues = true;
331 }
332
333 // Check dependencies
334 auto brokenRefs = Dependencies::ValidateAssetReferences(assetGuid);
335 if (!brokenRefs.empty()) {
336 result.healthy = false;
337 ValidationResult depIssue;
338 depIssue.assetGuid = assetGuid;
339 depIssue.status = ValidationStatus::BrokenDependencies;
340 depIssue.message = "Asset has broken dependencies";
341 for (const auto& broken : brokenRefs) {
342 depIssue.issues.push_back("Broken dependency: " + broken.reference);
343 }
344 result.issues.push_back(depIssue);
345 }
346
347 result.totalIssues = result.issues.size();
348 return result;

351bool RepairAsset(const std::string& assetGuid) {
352 ValidationResult validation = ValidateAsset(assetGuid);
353
354 if (validation.status == ValidationStatus::Valid) {
355 return true; // Nothing to repair
356 }
357
358 // Find asset
359 const AssetRecord* record = AssetRegistry::FindByGUID(assetGuid);
360 if (!record) {
361 return false;
362 }
363
364 // Try to repair based on status
365 switch (validation.status) {
366 case ValidationStatus::MissingMetadata:
367 case ValidationStatus::InvalidMetadata: {
368 // Create or repair metadata
369 Meta meta;
370 meta.guid = record->guid;
371 meta.type = record->type;
372 meta.sourceFile = record->absPath;
373 MetaDB::UpdateTimestamp(meta);
374 MetaDB::Save(record->absPath, meta);
375
376 // Clear cache
377 s_ValidationCache.erase(assetGuid);
378
379 Log::Info("Validation: Repaired metadata for: " + record->absPath);
380 return true;
381 }
382 case ValidationStatus::Missing: {
383 // Cannot repair missing files
384 return false;
385 }
386 case ValidationStatus::Corrupted: {
387 // Cannot repair corrupted files automatically
388 return false;
389 }
390 default:
391 return false;
392 }
