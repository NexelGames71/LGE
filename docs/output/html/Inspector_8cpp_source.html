199 std::string key = std::string(label) + "_" + std::to_string(m_SelectedObject->GetID());
200
201 if (texturePaths.find(key) == texturePaths.end()) {
202 texturePaths[key] = "";

204
205 char pathBuffer[512];
206 strncpy_s(pathBuffer, texturePaths[key].c_str(), sizeof(pathBuffer) - 1);

215 ImGui::SameLine();

217 if (ImGui::Button(("Load##" + std::string(label)).c_str())) {
218 std::string filepath = texturePaths[key];

222 // Try to load as regular image first
