61std::vector<std::string> ProjectManager::GetRecentProjects() {
62 std::vector<std::string> recentProjects;
63
64 std::string recentPath = GetRecentProjectsPath();
65 if (recentPath.empty() || !LxFile::Exists(recentPath)) {
66 return recentProjects;
67 }
68
69 try {
70 std::ifstream file(recentPath);
71 if (!file.is_open()) {
72 return recentProjects;
73 }
74
75 // Simple JSON parsing (basic implementation)
76 std::string line;
77 std::string content;
78 while (std::getline(file, line)) {
79 content += line;
80 }
81
82 // Find "recent_projects" array
83 size_t start = content.find("\"recent_projects\"");
84 if (start == std::string::npos) {
85 return recentProjects;
86 }
87
88 start = content.find("[", start);
89 if (start == std::string::npos) {
90 return recentProjects;
91 }
92
93 size_t end = content.find("]", start);
94 if (end == std::string::npos) {
95 return recentProjects;
96 }
97
98 // Extract project paths
99 std::string arrayContent = content.substr(start + 1, end - start - 1);
100 std::istringstream iss(arrayContent);
101 std::string item;
102
103 while (std::getline(iss, item, ',')) {
104 // Remove quotes and whitespace
105 item.erase(std::remove_if(item.begin(), item.end(),
106 [](char c) { return c == '"' || c == ' ' || c == '\n' || c == '\r' || c == '\t'; }),
107 item.end());
108
109 if (!item.empty() && LxFile::Exists(item)) {
110 recentProjects.push_back(item);
111 }
112 }
113
114 } catch (const std::exception& e) {
115 Log::Error("ProjectManager: Failed to load recent projects: " + std::string(e.what()));
116 }
117
118 return recentProjects;
119}

215bool ProjectManager::LoadProjectConfig(const std::string& projectPath, ProjectConfig& config) {
216 std::string configPath = GetProjectConfigPath(projectPath);
217
218 if (!LxFile::Exists(configPath)) {
219 return false;
220 }
221
222 try {
223 std::ifstream file(configPath);
224 if (!file.is_open()) {
225 return false;
226 }
227
228 // Simple JSON parsing (basic implementation)
229 std::string line;
230 std::string content;
231 while (std::getline(file, line)) {
232 content += line;
233 }
234
235 // Parse project name
236 size_t nameStart = content.find("\"name\"");
237 if (nameStart != std::string::npos) {
238 nameStart = content.find("\"", nameStart + 6);
239 if (nameStart != std::string::npos) {
240 size_t nameEnd = content.find("\"", nameStart + 1);
241 if (nameEnd != std::string::npos) {
242 config.name = content.substr(nameStart + 1, nameEnd - nameStart - 1);
243 }
244 }
245 }
246
247 // Parse graphics API
248 size_t apiStart = content.find("\"graphics_api\"");
249 if (apiStart != std::string::npos) {
250 apiStart = content.find("\"", apiStart + 14);
251 if (apiStart != std::string::npos) {
252 size_t apiEnd = content.find("\"", apiStart + 1);
253 if (apiEnd != std::string::npos) {
254 std::string apiStr = content.substr(apiStart + 1, apiEnd - apiStart - 1);
255 config.graphicsAPI = StringToGraphicsAPI(apiStr);
256 }
257 }
258 }
259
260 // Parse use_starter_content
261 size_t starterStart = content.find("\"use_starter_content\"");
262 if (starterStart != std::string::npos) {
263 size_t trueStart = content.find("true", starterStart);
264 if (trueStart != std::string::npos && trueStart < starterStart + 100) {
265 config.useStarterContent = true;
266 }
267 }
268
269 // Parse default scene
270 size_t sceneStart = content.find("\"default_scene\"");
271 if (sceneStart != std::string::npos) {
272 sceneStart = content.find("\"", sceneStart + 15);
273 if (sceneStart != std::string::npos) {
274 size_t sceneEnd = content.find("\"", sceneStart + 1);
275 if (sceneEnd != std::string::npos) {
276 config.defaultScene = content.substr(sceneStart + 1, sceneEnd - sceneStart - 1);
277 }
278 }
279 }
280
281 file.close();
282 return true;
283
284 } catch (const std::exception& e) {
285 Log::Error("ProjectManager: Failed to load project config: " + std::string(e.what()));
286 return false;

330bool ProjectManager::CreateProject(const ProjectSettings& settings) {
331 if (settings.name.empty() || settings.location.empty()) {
332 Log::Error("ProjectManager: Project name and location must be specified");
333 return false;
334 }
335
336 try {
337 // Create project directory
338 std::filesystem::path projectPath = std::filesystem::path(settings.location) / settings.name;
339
340 if (std::filesystem::exists(projectPath)) {
341 std::string errorMsg = "A project with the name \"" + settings.name + "\" already exists at:\n" + projectPath.string() + "\n\nPlease choose a different name or location.";
342 Log::Error("ProjectManager: " + errorMsg);
343 // Show user-friendly error message
344 #ifdef _WIN32
345 MessageBoxA(NULL, errorMsg.c_str(), "Project Already Exists", MB_OK | MB_ICONWARNING);
346 #endif
347 return false;
348 }
349
350 // Create project directory structure
351 std::filesystem::create_directories(projectPath);
352
353 // Create subdirectories
354 std::filesystem::create_directories(projectPath / "Assets");
355 std::filesystem::create_directories(projectPath / "Config");
356 std::filesystem::create_directories(projectPath / "Scripts");
357 std::filesystem::create_directories(projectPath / "Shaders");
358 std::filesystem::create_directories(projectPath / "Scenes");
359 std::filesystem::create_directories(projectPath / "Binaries");
360
361 // Initialize using LxFS (creates Assets, Packages, Cache, Settings)
362 std::string projectRoot = projectPath.string();
363 if (!LxFS::InitializeProject(projectRoot)) {
364 Log::Error("ProjectManager: Failed to initialize project structure");
365 return false;
366 }
367
368 // Create default scene
369 std::filesystem::path defaultScenePath = projectPath / "Scenes" / "DefaultScene.lscene";
370 std::ofstream sceneFile(defaultScenePath);
371 if (sceneFile.is_open()) {
372 sceneFile << "{\n";
373 sceneFile << " \"Name\": \"DefaultScene\",\n";
374 sceneFile << " \"Type\": \"Scene\",\n";
375 sceneFile << " \"GameObjects\": [],\n";
376 sceneFile << " \"Camera\": {\n";

378 sceneFile << " \"Target\": [0.0, 0.0, 0.0],\n";
379 sceneFile << " \"FOV\": 45.0\n";
380 sceneFile << " }\n";
381 sceneFile << "}\n";
382 sceneFile.close();
383 }
384
385 // Create project config
386 ProjectConfig projectConfig;
387 projectConfig.name = settings.name;
388 projectConfig.graphicsAPI = settings.graphicsAPI;
389 projectConfig.useStarterContent = settings.useStarterContent;
390 projectConfig.defaultScene = "Scenes/DefaultScene.lscene";
391
392 if (!SaveProjectConfig(projectRoot, projectConfig)) {
393 Log::Error("ProjectManager: Failed to save project config");
394 return false;
395 }
396
397 // Create .lumaproj file (project manifest)
398 std::filesystem::path lumaprojPath = projectPath / ".lumaproj";
399 std::ofstream lumaprojFile(lumaprojPath);
400 if (lumaprojFile.is_open()) {
401 // Get project GUID from manifest or generate one
402 ProjectManifest manifest;
403 std::string guid = "";
404 if (LxFS::LoadManifest(manifest)) {
405 guid = manifest.guid;
406 }
407 if (guid.empty()) {
408 guid = MetaDB::GenerateGUID();
409 }
410
411 lumaprojFile << "{\n";
412 lumaprojFile << " \"version\": 1,\n";
413 lumaprojFile << " \"name\": \"" << settings.name << "\",\n";
414 lumaprojFile << " \"engine_version\": \"1.0.0-alpha\",\n";
415 lumaprojFile << " \"guid\": \"" << guid << "\"\n";
416 lumaprojFile << "}\n";
417 lumaprojFile.close();
418 }
419
420 // Copy template files if template is specified
421 if (!settings.templateName.empty()) {
422 // Find template
423 auto templates = GetTemplates();
424 for (const auto& tmpl : templates) {
425 if (tmpl.name == settings.templateName) {
426 std::filesystem::path templatePath(tmpl.templatePath);
427 if (std::filesystem::exists(templatePath)) {
428 // Copy template files
429 // TODO: Implement template copying
430 Log::Info("ProjectManager: Template found: " + tmpl.name);
431 }
432 break;
433 }
434 }
435 }
436
437 // Re-initialize project to ensure it's properly set up for loading
438 if (!LxFS::InitializeProject(projectRoot)) {
439 Log::Error("ProjectManager: Failed to re-initialize project after creation");
440 return false;
441 }
442
443 // Validate project structure
444 if (!LxFS::ValidateProject()) {
445 Log::Error("ProjectManager: Project validation failed after creation");
446 return false;
447 }
448
449 // Verify project file exists
450 if (!std::filesystem::exists(lumaprojPath)) {
451 Log::Error("ProjectManager: Project file was not created: " + lumaprojPath.string());
452 return false;
453 }
454
455 // Save to recent projects
456 SaveRecentProject(lumaprojPath.string());
457
458 Log::Info("ProjectManager: Created and validated project: " + projectRoot);
459 return true;
460
461 } catch (const std::exception& e) {
462 Log::Error("ProjectManager: Failed to create project: " + std::string(e.what()));
463 return false;

467bool ProjectManager::OpenProject(const std::string& projectPath) {
468 if (projectPath.empty() || !LxFile::Exists(projectPath)) {
469 Log::Error("ProjectManager: Project path does not exist: " + projectPath);
470 return false;
471 }
472
473 try {
474 // Check if it's a .lumaproj file or directory
475 std::filesystem::path path(projectPath);
476 std::string projectRoot;
477
478 if (path.extension() == ".lumaproj") {
479 projectRoot = path.parent_path().string();
480 } else {
481 projectRoot = projectPath;
482 }
483
484 // Initialize project using LxFS
485 if (!LxFS::InitializeProject(projectRoot)) {
486 Log::Error("ProjectManager: Failed to initialize project");
487 return false;
488 }
489
490 // Load project config
491 ProjectConfig config;
492 if (LoadProjectConfig(projectRoot, config)) {
493 Log::Info("ProjectManager: Loaded project config for: " + config.name);
494 }
495
496 // Save to recent projects
497 std::string lumaprojPath = (std::filesystem::path(projectRoot) / ".lumaproj").string();
498 SaveRecentProject(lumaprojPath);
499
500 s_CurrentProjectPath = projectRoot;
501 s_IsProjectOpen = true;
502
503 Log::Info("ProjectManager: Opened project: " + projectRoot);
504 return true;
505
506 } catch (const std::exception& e) {
507 Log::Error("ProjectManager: Failed to open project: " + std::string(e.what()));
508 return false;
