28BenchmarkResult BenchmarkRunner::Run(const std::string& name,
29 std::function<void()> func,
30 const BenchmarkConfig& config) {
31 BenchmarkResult result;
32 result.name = name;
33 result.iterations = config.iterations;
34 result.sampleSize = config.samples;
35
36 if (m_Verbose || config.verbose) {
37 Log::Info("Benchmark: " + name);
38 Log::Info(" Warmup iterations: " + std::to_string(config.warmupIterations));
39 Log::Info(" Benchmark iterations: " + std::to_string(config.iterations));
40 Log::Info(" Samples per iteration: " + std::to_string(config.samples));
41 }
42
43 // Warmup
44 for (size_t i = 0; i < config.warmupIterations; ++i) {
45 func();
46 }
47
48 // Collect samples
49 std::vector<double> samples;
50 samples.reserve(config.iterations * config.samples);
51
52 for (size_t iter = 0; iter < config.iterations; ++iter) {
53 for (size_t sample = 0; sample < config.samples; ++sample) {
54 auto start = std::chrono::high_resolution_clock::now();
55 func();
56 auto end = std::chrono::high_resolution_clock::now();
57
58 auto duration = std::chrono::duration_cast<std::chrono::nanoseconds>(end - start);
59 double timeMs = duration.count() / 1000000.0;
60 samples.push_back(timeMs);
61 }
62 }
63
64 // Calculate statistics
65 CalculateStatistics(samples, result);
66
67 // Check if benchmark passed target
68 if (config.targetTimeMs > 0.0) {
69 result.passed = result.averageTimeMs <= config.targetTimeMs;
70 if (m_Verbose || config.verbose) {
71 if (result.passed) {
72 Log::Info(" ✓ PASSED (target: " + std::to_string(config.targetTimeMs) + "ms)");
73 } else {
74 Log::Warn(" ✗ FAILED (target: " + std::to_string(config.targetTimeMs) +
75 "ms, actual: " + std::to_string(result.averageTimeMs) + "ms)");
76 }
77 }
78 } else {
79 result.passed = true; // No target specified
80 }
81
82 m_Results.push_back(result);
83
84 if (m_Verbose || config.verbose) {
85 std::cout << " Average: " << std::fixed << std::setprecision(3)
86 << result.averageTimeMs << "ms" << std::endl;
87 std::cout << " Min: " << result.minTimeMs << "ms" << std::endl;
88 std::cout << " Max: " << result.maxTimeMs << "ms" << std::endl;
89 std::cout << " Median: " << result.medianTimeMs << "ms" << std::endl;
90 }

108void BenchmarkRunner::CalculateStatistics(std::vector<double>& samples, BenchmarkResult& result) {
109 if (samples.empty()) {
110 return;
111 }
112
113 // Sort for median calculation
114 std::sort(samples.begin(), samples.end());
115
116 // Calculate average
117 double sum = std::accumulate(samples.begin(), samples.end(), 0.0);
118 result.averageTimeMs = sum / samples.size();
119
120 // Min and max
121 result.minTimeMs = samples.front();
122 result.maxTimeMs = samples.back();
123
124 // Median
125 size_t size = samples.size();
126 if (size % 2 == 0) {
127 result.medianTimeMs = (samples[size / 2 - 1] + samples[size / 2]) / 2.0;
128 } else {
129 result.medianTimeMs = samples[size / 2];
130 }
131}
