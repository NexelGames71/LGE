24// Helper: Get file extension
25static std::string GetExtension(const std::string& filePath) {
26 std::filesystem::path path(filePath);
27 std::string ext = path.extension().string();
28 if (!ext.empty() && ext[0] == '.') {
29 ext = ext.substr(1);
30 }
31 std::transform(ext.begin(), ext.end(), ext.begin(), ::tolower);
32 return ext;
33}
34
35// Helper: Check if extension is allowed
36static bool IsExtensionAllowed(const std::string& ext, const std::vector<std::string>& allowed) {
37 if (allowed.empty()) {
38 return true; // No filter, allow all
39 }
40
41 std::string lowerExt = ext;
42 std::transform(lowerExt.begin(), lowerExt.end(), lowerExt.begin(), ::tolower);
43
44 for (const auto& allowedExt : allowed) {
45 std::string lowerAllowed = allowedExt;
46 std::transform(lowerAllowed.begin(), lowerAllowed.end(), lowerAllowed.begin(), ::tolower);
47 if (lowerExt == lowerAllowed || lowerExt == "." + lowerAllowed) {
48 return true;
49 }
50 }
51
52 return false;
53}

55ImportResult ImportFile(const std::string& sourcePath,
56 const ImportOptions& options,
57 ImportProgressCallback progressCallback) {
58 ImportResult result;
59 result.success = false;
60
61 auto startTime = std::chrono::high_resolution_clock::now();
62
63 // Check if source exists
64 if (!LxFile::Exists(sourcePath)) {
65 result.errorMessage = "Source file does not exist: " + sourcePath;
66 return result;
67 }
68
69 // Check if it's a file
70 if (!LxFile::IsFile(sourcePath)) {
71 result.errorMessage = "Source path is not a file: " + sourcePath;
72 return result;
73 }
74
75 // Check extension filter
76 std::string ext = GetExtension(sourcePath);
77 if (!IsExtensionAllowed(ext, options.allowedExtensions)) {
78 result.errorMessage = "File extension not allowed: " + ext;
79 result.success = true; // Not an error, just skipped
80 return result;
81 }
82
83 // Get file size
84 result.fileSize = LxFile::GetFileSize(sourcePath);
85
86 // Determine target directory
87 std::string targetDir = options.targetDirectory;
88 if (targetDir.empty()) {
89 // Use Assets directory
90 targetDir = LxPath::GetRootPath(LxPath::Root::Assets);
91 }
92
93 // Get filename
94 std::string filename = LxPath::GetFilename(sourcePath);
95 std::string targetPath = LxPath::Join(targetDir, filename);
96
97 // Check if target exists
98 if (LxFile::Exists(targetPath) && !options.overwriteExisting) {
99 result.errorMessage = "Target file already exists: " + targetPath;
100 return result;
101 }
102
103 // Call progress callback
104 if (progressCallback) {
105 progressCallback(1, 1, filename, 0.5f);
106 }
107
108 // Use FSOps::ImportFile
109 IOError err = FSOps::ImportFile(sourcePath, LxPath::Root::Assets,
110 LxPath::ToRelative(targetDir));
111 if (err != IOError::None) {
112 result.errorMessage = "Import failed: " + std::to_string(static_cast<int>(err));
113 return result;
114 }
115
116 // Get imported asset info
117 const AssetRecord* record = AssetRegistry::FindByPath(targetPath);
118 if (record) {
119 result.assetPath = record->absPath;
120 result.assetGuid = record->guid;
121 } else {
122 result.assetPath = targetPath;
123 }
124
125 // Generate thumbnail if requested
126 if (options.generateThumbnails && record) {
127 Thumbs::EnqueueBuild(record->guid, record->type, record->absPath);
128 }
129
130 // Validate dependencies if requested
131 if (options.validateDependencies && record) {
132 auto brokenRefs = Dependencies::ValidateAssetReferences(record->guid);
133 if (!brokenRefs.empty()) {
134 Log::Warn("ImportExport: Found " + std::to_string(brokenRefs.size()) +
135 " broken references in imported asset: " + targetPath);
136 }
137 }
138
139 auto endTime = std::chrono::high_resolution_clock::now();
140 result.importTime = std::chrono::duration_cast<std::chrono::milliseconds>(
141 endTime - startTime).count();
142
143 result.success = true;
144
145 // Update statistics
146 s_ImportStats.totalImported++;
147 s_ImportStats.totalBytes += result.fileSize;
148 s_ImportStats.totalTime += result.importTime;
149
150 // Call progress callback
151 if (progressCallback) {
152 progressCallback(1, 1, filename, 1.0f);
153 }
154
155 Log::Info("ImportExport: Imported: " + sourcePath + " -> " + targetPath);
156 return result;
157}

159std::vector<ImportResult> ImportFiles(const std::vector<std::string>& sourcePaths,
160 const ImportOptions& options,
161 ImportProgressCallback progressCallback) {
162 std::vector<ImportResult> results;
163 results.reserve(sourcePaths.size());
164
165 size_t total = sourcePaths.size();
166 size_t current = 0;
167
168 for (const auto& sourcePath : sourcePaths) {
169 current++;
170
171 // Call progress callback
172 if (progressCallback) {
173 float progress = static_cast<float>(current - 1) / total;
174 progressCallback(current - 1, total, LxPath::GetFilename(sourcePath), progress);
175 }
176
177 ImportResult result = ImportFile(sourcePath, options, nullptr);
178 results.push_back(result);
179
180 if (!result.success) {
181 s_ImportStats.totalFailed++;
182 } else if (!result.errorMessage.empty() &&
183 result.errorMessage.find("not allowed") != std::string::npos) {
184 s_ImportStats.totalSkipped++;
185 }
186 }
187
188 // Final progress callback
189 if (progressCallback) {
190 progressCallback(total, total, "", 1.0f);
191 }
192
193 Log::Info("ImportExport: Batch import complete: " + std::to_string(results.size()) + " files");
194 return results;
195}

214ExportResult ExportAsset(const std::string& assetGuid,
215 const std::string& outputPath,
216 const ExportOptions& options,
217 ExportProgressCallback progressCallback) {
218 ExportResult result;
219 result.success = false;
220
221 auto startTime = std::chrono::high_resolution_clock::now();
222
223 // Find asset
224 const AssetRecord* record = AssetRegistry::FindByGUID(assetGuid);
225 if (!record) {
226 result.errorMessage = "Asset not found: " + assetGuid;
227 return result;
228 }
229
230 // Check if asset file exists
231 if (!LxFile::Exists(record->absPath)) {
232 result.errorMessage = "Asset file does not exist: " + record->absPath;
233 return result;
234 }
235
236 // Call progress callback
237 if (progressCallback) {
238 progressCallback(1, 1, record->name, 0.5f);
239 }
240
241 // Use ExporterRegistry if format is specified
242 if (!options.format.empty()) {
243 // Get asset type
244 AssetType assetType = StringToAssetType(record->type);
245
246 // Get exporter
247 auto exporter = ExporterRegistry::Get().GetExporterForAsset(assetType, options.format);
248 if (exporter) {
249 // Load metadata
250 auto meta = MetaDB::Load(record->absPath);
251 if (meta.has_value()) {
252 // Convert Meta to LxAssetMeta (simplified)
253 LxAssetMeta assetMeta;
254 assetMeta.path = record->absPath;
255 assetMeta.name = record->name;
256 assetMeta.type = assetType;
257 assetMeta.guid = record->guid;
258
259 // Export using exporter
260 if (exporter->Export(record->absPath, outputPath, assetMeta)) {
261 result.success = true;
262 result.outputPath = outputPath;
263 } else {
264 result.errorMessage = "Export failed using exporter";
265 }
266 } else {
267 result.errorMessage = "Metadata not found for asset";
268 }
269 } else {
270 // Fall back to simple copy
271 IOError err = LxFile::Copy(record->absPath, outputPath, true);
272 if (err == IOError::None) {
273 result.success = true;
274 result.outputPath = outputPath;
275 } else {
276 result.errorMessage = "Export failed: " + std::to_string(static_cast<int>(err));
277 }
278 }
279 } else {
280 // Simple copy
281 IOError err = LxFile::Copy(record->absPath, outputPath, true);
282 if (err == IOError::None) {
283 result.success = true;
284 result.outputPath = outputPath;
285 } else {
286 result.errorMessage = "Export failed: " + std::to_string(static_cast<int>(err));
287 }
288 }
289
290 if (result.success) {
291 result.fileSize = LxFile::GetFileSize(outputPath);
292 }
293
294 auto endTime = std::chrono::high_resolution_clock::now();
295 result.exportTime = std::chrono::duration_cast<std::chrono::milliseconds>(
296 endTime - startTime).count();
297
298 // Call progress callback
299 if (progressCallback) {
300 progressCallback(1, 1, record->name, 1.0f);
301 }
302
303 if (result.success) {
304 Log::Info("ImportExport: Exported: " + record->absPath + " -> " + outputPath);
305 }
306
307 return result;
308}
