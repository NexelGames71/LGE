786 void SaveScene() {
787 if (m_CurrentProjectPath.empty()) {
788 LGE::Log::Error("ExampleApp: Cannot save scene - no project loaded");
789 return;
790 }
791
792 std::string sceneName = std::string(m_SceneNameBuffer);
793 if (sceneName.empty()) {
794 LGE::Log::Error("ExampleApp: Cannot save scene - scene name is empty");
795 return;
796 }
797
798 // Ensure scene name doesn't have extension
799 if (sceneName.find(".scene") != std::string::npos) {
800 sceneName = sceneName.substr(0, sceneName.find(".scene"));
801 }
802
803 // Create Scenes directory if it doesn't exist
804 std::filesystem::path scenesPath = std::filesystem::path(m_CurrentProjectPath) / "Assets" / "Scenes";
805 if (!std::filesystem::exists(scenesPath)) {
806 std::filesystem::create_directories(scenesPath);
807 }
808
809 // Create scene file path
810 std::filesystem::path sceneFilePath = scenesPath / (sceneName + ".scene");
811
812 // If the scene name is "SampleScene" and it already exists, prompt to rename
813 // Otherwise, save the scene
814 try {
815 // Generate scene content (for now, just save the current scene structure)
816 std::ofstream sceneFile(sceneFilePath);
817 if (sceneFile.is_open()) {
818 sceneFile << "{\n";
819 sceneFile << " \"Name\": \"" << sceneName << "\",\n";
820 sceneFile << " \"Type\": \"Scene\",\n";
821 sceneFile << " \"GameObjects\": [],\n";
822 sceneFile << " \"Camera\": {\n";
823 sceneFile << " \"Position\": [0.0, 0.0, 3.0],\n";
824 sceneFile << " \"Target\": [0.0, 0.0, 0.0],\n";
825 sceneFile << " \"FOV\": 45.0\n";
826 sceneFile << " },\n";
827 sceneFile << " \"Lighting\": {\n";
828 sceneFile << " \"DirectionalLight\": {\n";
829 sceneFile << " \"Direction\": [0.3, -0.8, 0.5],\n";
830 sceneFile << " \"Color\": [1.0, 0.98, 0.95],\n";
831 sceneFile << " \"Intensity\": 0.8\n";
832 sceneFile << " }\n";
833 sceneFile << " }\n";
834 sceneFile << "}\n";
835 sceneFile.close();
836
837 // Update current scene path
838 m_CurrentScenePath = std::filesystem::absolute(sceneFilePath).string();
839 std::replace(m_CurrentScenePath.begin(), m_CurrentScenePath.end(), '\\', '/');
840
841 LGE::Log::Info("ExampleApp: Saved scene: " + m_CurrentScenePath);
842
843 // Rescan assets if asset manager is available
844 if (m_AssetManager) {
845 m_AssetManager->ScanAssets();
846 }
847 } else {
848 LGE::Log::Error("ExampleApp: Failed to create scene file: " + sceneFilePath.string());
849 }
850 } catch (const std::exception& e) {
851 LGE::Log::Error("ExampleApp: Failed to save scene: " + std::string(e.what()));
852 }
853 }
