21struct PairHash {
22 std::size_t operator()(const std::pair<int, int>& p) const {
23 return std::hash<int>()(p.first) ^ (std::hash<int>()(p.second) << 1);

59// Migrate metadata to latest version
60bool MigrateMetadata(Meta& meta) {
61 if (!NeedsMigration(meta)) {
62 return true; // Already at latest version
63 }
64
65 int currentVersion = meta.version;
66 int targetVersion = CURRENT_METADATA_VERSION;
67
68 Log::Info("Migration: Migrating metadata from v" + std::to_string(currentVersion) +
69 " to v" + std::to_string(targetVersion));
70
71 // Apply migrations step by step
72 while (currentVersion < targetVersion) {
73 int nextVersion = currentVersion + 1;
74
75 // Find migration function
76 auto it = s_Migrations.find({currentVersion, nextVersion});
77 if (it == s_Migrations.end()) {
78 // Try to find a direct migration to target version
79 it = s_Migrations.find({currentVersion, targetVersion});
80 if (it == s_Migrations.end()) {
81 Log::Error("Migration: No migration function found from v" +
82 std::to_string(currentVersion) + " to v" + std::to_string(nextVersion));
83 return false;
84 }
85 }
86
87 // Apply migration
88 try {
89 meta = it->second(meta);
90 meta.version = nextVersion;
91 currentVersion = nextVersion;
92
93 Log::Info("Migration: Successfully migrated metadata to v" + std::to_string(currentVersion));
94 } catch (const std::exception& e) {
95 Log::Error("Migration: Failed to migrate metadata: " + std::string(e.what()));
96 return false;
97 }
98 }
99
100 // Update timestamp after migration
101 MetaDB::UpdateTimestamp(meta);
102
103 return true;
104}

144// Migrate all metadata files in a directory
145size_t MigrateAllMetadata(const std::string& absPath,
146 std::function<void(size_t, size_t, const std::string&)> progressCallback) {
147 if (absPath.empty() || !LxFile::IsDirectory(absPath)) {
148 return 0;
149 }
150
151 size_t migratedCount = 0;
152 size_t totalCount = 0;
153 std::vector<std::string> metadataFiles;
154
155 try {
156 // Collect all metadata files
157 for (const auto& entry : std::filesystem::recursive_directory_iterator(absPath)) {
158 if (entry.is_regular_file()) {
159 std::string path = entry.path().string();
160 if (path.ends_with(".lumameta")) {
161 metadataFiles.push_back(path);
162 totalCount++;
163 }
164 }
165 }
166
167 // Migrate each metadata file
168 for (size_t i = 0; i < metadataFiles.size(); ++i) {
169 const std::string& metaPath = metadataFiles[i];
170
171 // Call progress callback
172 if (progressCallback) {
173 progressCallback(i, totalCount, metaPath);
174 }
175
176 // Load metadata
177 auto meta = MetaDB::Load(metaPath);
178 if (!meta.has_value()) {
179 Log::Warn("Migration: Failed to load metadata: " + metaPath);
180 continue;
181 }
182
183 // Check if migration is needed
184 if (!NeedsMigration(*meta)) {
185 continue; // Already at latest version
186 }
187
188 // Migrate metadata
189 Meta migratedMeta = *meta;
190 if (!MigrateMetadata(migratedMeta)) {
191 Log::Error("Migration: Failed to migrate metadata: " + metaPath);
192 continue;
193 }
194
195 // Save migrated metadata
196 MetaDB::Save(metaPath, migratedMeta);
197
198 migratedCount++;
199
200 // Update registry if asset exists
201 std::string assetPath = metaPath;
202 if (assetPath.ends_with(".lumameta")) {
203 assetPath = assetPath.substr(0, assetPath.length() - 9); // Remove ".lumameta"
204 }
205 if (LxFile::Exists(assetPath)) {
206 AssetRegistry::AddOrUpdate(assetPath);
207 }
208 }
209
210 Log::Info("Migration: Migrated " + std::to_string(migratedCount) +
211 " of " + std::to_string(totalCount) + " metadata files");
212 } catch (const std::exception& e) {
213 Log::Error("Migration: Failed to migrate metadata files: " + std::string(e.what()));
214 }
215
216 return migratedCount;
217}
