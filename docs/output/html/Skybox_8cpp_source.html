142 // Remove translation from view matrix (set translation to 0)
143 viewMatrix.GetData()[12] = 0.0f;

148 m_Shader->SetUniformMat4("u_ViewProjection", viewProjection.GetData());

152

157 m_Shader->SetUniform1i("u_UseTexture", 1);
