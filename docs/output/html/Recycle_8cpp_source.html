29static bool ReadDeleteInfo(const std::string& binPath, Entry& entry) {
30 std::string infoPath = binPath + ".info";
31
32 if (!LxFile::Exists(infoPath)) {
33 return false;
34 }
35
36 std::string jsonContent;
37 IOError err = LxFile::ReadAll(infoPath, jsonContent);
38 if (err != IOError::None) {
39 return false;
40 }
41
42 // Simple JSON parsing
43 try {
44 // Extract originalVirtualPath
45 size_t pos = jsonContent.find("\"originalVirtualPath\"");
46 if (pos != std::string::npos) {
47 pos = jsonContent.find(':', pos);
48 if (pos != std::string::npos) {
49 size_t start = jsonContent.find('"', pos) + 1;
50 size_t end = jsonContent.find('"', start);
51 entry.originalVirtualPath = jsonContent.substr(start, end - start);
52 }
53 }
54
55 // Extract originalAbsPath
56 pos = jsonContent.find("\"originalAbsPath\"");
57 if (pos != std::string::npos) {
58 pos = jsonContent.find(':', pos);
59 if (pos != std::string::npos) {
60 size_t start = jsonContent.find('"', pos) + 1;
61 size_t end = jsonContent.find('"', start);
62 entry.originalAbsPath = jsonContent.substr(start, end - start);
63 }
64 }
65
66 // Extract originalName
67 pos = jsonContent.find("\"originalName\"");
68 if (pos != std::string::npos) {
69 pos = jsonContent.find(':', pos);
70 if (pos != std::string::npos) {
71 size_t start = jsonContent.find('"', pos) + 1;
72 size_t end = jsonContent.find('"', start);
73 entry.originalName = jsonContent.substr(start, end - start);
74 }
75 }
76
77 // Extract deletedAt
78 pos = jsonContent.find("\"deletedAt\"");
79 if (pos != std::string::npos) {
80 pos = jsonContent.find(':', pos);
81 if (pos != std::string::npos) {
82 size_t end = jsonContent.find_first_of(",}", pos);
83 std::string val = jsonContent.substr(pos + 1, end - pos - 1);
84 val.erase(0, val.find_first_not_of(" \t"));
85 val.erase(val.find_last_not_of(" \t") + 1);
86 entry.deletedAt = std::stoull(val);
87 }
88 }
89
90 entry.binPath = binPath;
91 return true;
92 } catch (const std::exception& e) {
93 Log::Error("Recycle: Failed to parse delete info: " + std::string(e.what()));
94 return false;
95 }

114// List all entries in recycle bin
115std::vector<Entry> List() {
116 std::vector<Entry> entries;
117
118 std::string recycleBin = GetRecycleBinPath();
119 if (recycleBin.empty() || !LxFile::Exists(recycleBin)) {
120 return entries;
121 }
122
123 try {
124 for (const auto& entry : std::filesystem::directory_iterator(recycleBin)) {
125 if (entry.is_regular_file()) {
126 std::string path = entry.path().string();
127 std::replace(path.begin(), path.end(), '\\', '/');
128
129 // Skip .info files
130 if (path.length() > 5 && path.substr(path.length() - 5) == ".info") {
131 continue;
132 }
133
134 // Read delete info
135 Entry recycleEntry;
136 if (ReadDeleteInfo(path, recycleEntry)) {
137 entries.push_back(recycleEntry);
138 }
139 }
140 }
141
142 // Sort by deletion time (newest first)
143 std::sort(entries.begin(), entries.end(),
144 [](const Entry& a, const Entry& b) {
145 return a.deletedAt > b.deletedAt;
146 });
147 } catch (const std::exception& e) {
148 Log::Error("Recycle: Failed to list recycle bin: " + std::string(e.what()));
149 }
150
151 return entries;
152}

154// Restore an entry from recycle bin
155IOError Restore(const Entry& entry, bool overwrite) {
156 if (entry.binPath.empty()) {
157 return IOError::Invalid;
158 }
159
160 // Check if entry exists in recycle bin
161 if (!LxFile::Exists(entry.binPath)) {
162 return IOError::NotFound;
163 }
164
165 // Resolve original path
166 std::string restorePath = entry.originalAbsPath;
167
168 // If originalAbsPath is empty, try to resolve from virtual path
169 if (restorePath.empty() && !entry.originalVirtualPath.empty()) {
170 auto resolved = VFS::ResolveAbsolute(entry.originalVirtualPath);
171 if (resolved.has_value()) {
172 restorePath = *resolved;
173 }
174 }
175
176 if (restorePath.empty()) {
177 Log::Error("Recycle: Cannot restore - original path is empty");
178 return IOError::Invalid;
179 }
180
181 // Check if destination exists
182 if (!overwrite && LxFile::Exists(restorePath)) {
183 return IOError::Exists;
184 }
185
186 // Ensure parent directory exists
187 std::string parentDir = LxPath::GetParent(restorePath);
188 if (!parentDir.empty() && !LxFile::Exists(parentDir)) {
189 IOError err = LxFile::CreateDirectory(parentDir);
190 if (err != IOError::None) {
191 return err;
192 }
193 }
194
195 // Restore file
196 IOError err = LxFile::Move(entry.binPath, restorePath, overwrite);
197 if (err != IOError::None) {
198 return err;
199 }
200
201 // Restore metadata if it exists
202 std::string binMetaPath = entry.binPath + ".lumameta";
203 std::string restoreMetaPath = restorePath + ".lumameta";
204 if (LxFile::Exists(binMetaPath)) {
205 LxFile::Move(binMetaPath, restoreMetaPath, overwrite);
206 }
207
208 // Delete .info file
209 std::string infoPath = entry.binPath + ".info";
210 if (LxFile::Exists(infoPath)) {
211 LxFile::Delete(infoPath, true); // Permanent delete
212 }
213
214 // Update registry
215 AssetRegistry::AddOrUpdate(restorePath);
216
217 Log::Info("Recycle: Restored: " + entry.binPath + " -> " + restorePath);
218 return IOError::None;
219}

249// Purge all entries older than specified days
250size_t PurgeOlderThanDays(int days) {
251 if (days < 0) {
252 return 0;
253 }
254
255 auto entries = List();
256 size_t purgedCount = 0;
257
258 auto now = std::chrono::system_clock::now();
259 auto cutoffTime = now - std::chrono::hours(24 * days);
260 uint64_t cutoffTimestamp = std::chrono::duration_cast<std::chrono::seconds>(
261 cutoffTime.time_since_epoch()
262 ).count();
263
264 for (const auto& entry : entries) {
265 if (entry.deletedAt < cutoffTimestamp) {
266 IOError err = Purge(entry);
267 if (err == IOError::None) {
268 purgedCount++;
269 }
270 }
271 }
272
273 if (purgedCount > 0) {
274 Log::Info("Recycle: Purged " + std::to_string(purgedCount) + " entries older than " + std::to_string(days) + " days");
275 }
276
