213BackupResult CreateBackup(const BackupOptions& options) {
214 BackupResult result;
215
216 auto startTime = std::chrono::high_resolution_clock::now();
217
218 if (!s_IsInitialized) {
219 result.errorMessage = "Backup system not initialized";
220 return result;
221 }
222
223 // Generate backup ID
224 std::string backupId = GenerateBackupId();
225 result.backupId = backupId;
226
227 // Get backup path
228 std::string backupPath = GetBackupPath(backupId, options);
229
230 // Create backup entry
231 BackupEntry entry;
232 entry.backupId = backupId;
233 entry.name = options.name.empty() ? "Backup " + backupId : options.name;
234 entry.timestamp = GetCurrentTime();
235 entry.backupPath = backupPath;
236 entry.isComplete = false;
237
238 // Collect assets to backup
239 std::vector<std::string> filesToBackup;
240
241 if (options.includeAssets) {
242 auto allAssets = AssetRegistry::GetAllAssets();
243 for (const auto& asset : allAssets) {
244 if (LxFile::Exists(asset.absPath)) {
245 filesToBackup.push_back(asset.absPath);
246 result.assetsBackedUp++;
247 }
248 }
249 }
250
251 if (options.includeMetadata) {
252 // Add metadata files
253 auto allAssets = AssetRegistry::GetAllAssets();
254 for (const auto& asset : allAssets) {
255 std::string metaPath = MetaDB::MetaPathFor(asset.absPath);
256 if (LxFile::Exists(metaPath)) {
257 filesToBackup.push_back(metaPath);
258 }
259 }
260 }
261
262 if (options.includeDependencies) {
263 // Add dependencies (could be enhanced to include dependency files)
264 // For now, we'll just note that dependencies are included
265 }
266
267 entry.assetCount = result.assetsBackedUp;
268
269 // Create backup
270 if (options.compress) {
271 // Compress backup
272 CompressionOptions compOptions;
273 compOptions.format = Compression::GetFormatFromExtension("." + options.compressionFormat);
274 compOptions.level = CompressionLevel::Balanced;
275 compOptions.overwriteExisting = true;
276
277 CompressionResult compResult = Compression::CompressFiles(filesToBackup, backupPath, compOptions);
278 if (!compResult.success) {
279 result.errorMessage = "Compression failed: " + compResult.errorMessage;
280 entry.errorMessage = result.errorMessage;
281 SaveBackupMeta(backupId, entry);
282 return result;
283 }
284
285 entry.totalSize = compResult.compressedSize;
286 result.totalSize = compResult.compressedSize;
287 } else {
288 // Simple copy (create directory structure)
289 std::string backupDir = LxPath::GetParent(backupPath);
290 if (!LxFile::Exists(backupDir)) {
291 LxFile::CreateDirectory(backupDir);
292 }
293
294 // For non-compressed backups, we'd need to create a directory structure
295 // For now, we'll use compression as the default
296 result.errorMessage = "Non-compressed backups not yet implemented";
297 entry.errorMessage = result.errorMessage;
298 SaveBackupMeta(backupId, entry);
299 return result;
300 }
301
302 // Calculate checksum
303 entry.checksum = Validation::GetChecksum(backupPath, false);
304 entry.isComplete = true;
305
306 // Save backup metadata
307 if (!SaveBackupMeta(backupId, entry)) {
308 result.errorMessage = "Failed to save backup metadata";
309 return result;
310 }
311
312 // Store in memory
313 s_Backups[backupId] = entry;
314
315 auto endTime = std::chrono::high_resolution_clock::now();
316 result.backupTime = std::chrono::duration_cast<std::chrono::milliseconds>(
317 endTime - startTime).count();
318
319 result.success = true;
320 s_LastBackupTime = GetCurrentTime();
321
322 Log::Info("Backup: Created backup " + backupId + " (" +
323 std::to_string(result.assetsBackedUp) + " assets, " +
324 std::to_string(result.totalSize) + " bytes)");
325
326 return result;
327}

346RestoreResult RestoreBackup(const std::string& backupId,
347 const RestoreOptions& options) {
348 RestoreResult result;
349
350 auto startTime = std::chrono::high_resolution_clock::now();
351
352 if (!s_IsInitialized) {
353 result.errorMessage = "Backup system not initialized";
354 return result;
355 }
356
357 // Get backup entry
358 auto backup = GetBackup(backupId);
359 if (!backup.has_value()) {
360 result.errorMessage = "Backup not found: " + backupId;
361 return result;
362 }
363
364 if (!backup->isComplete) {
365 result.errorMessage = "Backup is incomplete: " + backupId;
366 return result;
367 }
368
369 // Check if backup file exists
370 if (!LxFile::Exists(backup->backupPath)) {
371 result.errorMessage = "Backup file does not exist: " + backup->backupPath;
372 return result;
373 }
374
375 // Determine restore path
376 std::string restorePath = options.targetPath.empty() ?
377 LxFS::Paths().assets : options.targetPath;
378
379 // Decompress backup
380 CompressionFormat format = Compression::GetFormatFromExtension(backup->backupPath);
381 if (format != CompressionFormat::None) {
382 DecompressionResult decompResult = Compression::DecompressArchive(
383 backup->backupPath, restorePath, options.overwriteExisting);
384
385 if (!decompResult.success) {
386 result.errorMessage = "Decompression failed: " + decompResult.errorMessage;
387 return result;
388 }
389
390 result.assetsRestored = backup->assetCount;
391 } else {
392 result.errorMessage = "Unsupported backup format";
393 return result;
394 }
395
396 auto endTime = std::chrono::high_resolution_clock::now();
397 result.restoreTime = std::chrono::duration_cast<std::chrono::milliseconds>(
398 endTime - startTime).count();
399
400 result.success = true;
401
402 Log::Info("Backup: Restored backup " + backupId + " (" +
403 std::to_string(result.assetsRestored) + " assets)");
404
405 return result;
406}
