35std::shared_ptr<Texture> ThumbnailGenerator::GenerateThumbnail(const std::string& assetPath, const LxAssetMeta& meta) {
36 // Check if thumbnail already exists in cache
37 auto cached = LoadThumbnailFromCache(assetPath);
38 if (cached) {
39 return cached;
40 }
41
42 std::shared_ptr<Texture> thumbnail = nullptr;
43
44 // Generate thumbnail based on asset type
45 switch (meta.type) {
46 case AssetType::Texture2D:
47 case AssetType::TextureCube:
48 thumbnail = GenerateTextureThumbnail(assetPath);
49 break;
50
51 case AssetType::Model:
52 thumbnail = GenerateModelThumbnail(assetPath);
53 break;
54
55 case AssetType::Folder:
56 thumbnail = GenerateDefaultThumbnail(AssetType::Folder);
57 break;
58
59 case AssetType::Material:
60 thumbnail = GenerateMaterialThumbnail(assetPath, meta);
61 break;
62
63 default:
64 thumbnail = GenerateDefaultThumbnail(meta.type);
65 break;
66 }
67
68 // Save to cache if generated
69 if (thumbnail) {
70 SaveThumbnailToCache(assetPath, thumbnail);
71 }
72
73 return thumbnail;
74}
