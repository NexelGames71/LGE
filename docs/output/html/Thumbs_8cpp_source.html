122// Build thumbnail synchronously
123bool Build(const std::string& guid, std::string_view type, const std::string& assetAbsPath) {
124 if (guid.empty() || assetAbsPath.empty()) {
125 return false;
126 }
127
128 EnsureThumbnailGenerator();
129
130 // Check if asset exists
131 if (!LxFile::Exists(assetAbsPath)) {
132 Log::Error("Thumbs: Cannot build thumbnail - asset not found: " + assetAbsPath);
133 return false;
134 }
135
136 // Load metadata if available
137 auto meta = MetaDB::Load(assetAbsPath);
138
139 // Convert type string to AssetType
140 AssetType assetType = AssetType::Unknown;
141 std::string typeStr(type);
142 if (typeStr == "Texture2D" || typeStr == "Texture") {
143 assetType = AssetType::Texture2D;
144 } else if (typeStr == "Mesh" || typeStr == "Model") {
145 assetType = AssetType::Model;
146 } else if (typeStr == "Material") {
147 assetType = AssetType::Material;
148 } else if (typeStr == "Folder") {
149 assetType = AssetType::Folder;
150 }
151
152 // Create LxAssetMeta for ThumbnailGenerator
153 LxAssetMeta lxMeta;
154 lxMeta.type = assetType;
155 if (meta.has_value()) {
156 lxMeta.guid = meta->guid;
157 } else {
158 lxMeta.guid = guid;
159 }
160
161 // Generate thumbnail using ThumbnailGenerator
162 // This will use the existing cache mechanism (filename-based)
163 auto thumbnail = s_ThumbnailGenerator->GenerateThumbnail(assetAbsPath, lxMeta);
164 if (!thumbnail) {
165 Log::Warn("Thumbs: Failed to generate thumbnail for: " + assetAbsPath);
166 return false;
167 }
168
169 // Get the thumbnail path that ThumbnailGenerator uses (filename-based)
170 std::string generatorPath = s_ThumbnailGenerator->GetThumbnailPath(assetAbsPath);
171
172 // Get our GUID-based path
173 std::string thumbPath = Path(guid);
174 if (thumbPath.empty()) {
175 return false;
176 }
177
178 // Ensure directory exists
179 std::string thumbDir = LxPath::GetParent(thumbPath);
180 if (!thumbDir.empty() && !LxFile::Exists(thumbDir)) {
181 LxFile::CreateDirectory(thumbDir);
182 }
183
184 // If ThumbnailGenerator saved to disk, copy to GUID-based location
185 // Otherwise, we'll just cache the texture in memory
186 if (LxFile::Exists(generatorPath)) {
187 // Copy from generator cache to GUID-based location
188 IOError err = LxFile::Copy(generatorPath, thumbPath, true); // Overwrite
189 if (err != IOError::None) {
190 Log::Warn("Thumbs: Failed to copy thumbnail to GUID location: " + thumbPath);
191 }
192 }
193
194 // Cache the texture
195 {
196 std::lock_guard<std::mutex> lock(s_CacheMutex);
197 s_ThumbnailCache[guid] = thumbnail;
198 }
199
200 // Publish event
201 EditorBus::Publish(EditorBus::Topic::Thumbnail, "Built", guid);
202
203 Log::Info("Thumbs: Built thumbnail for: " + guid);
204 return true;
205}

207// Enqueue thumbnail build task
208void EnqueueBuild(const std::string& guid, std::string_view type, const std::string& assetAbsPath) {
209 if (guid.empty() || assetAbsPath.empty()) {
210 return;
211 }
212
213 // Check if already exists and is up-to-date
214 if (Exists(guid) && !NeedsRebuild(guid, assetAbsPath)) {
215 return; // Already exists and is current
216 }
217
218 BuildTask task(guid, std::string(type), assetAbsPath);
219
220 {
221 std::lock_guard<std::mutex> lock(s_BuildQueueMutex);
222 s_BuildQueue.push(task);
223 }
224
225 // Publish event
226 EditorBus::Publish(EditorBus::Topic::Thumbnail, "Enqueued", guid);
227}

229// Check if thumbnail needs to be rebuilt
230bool NeedsRebuild(const std::string& guid, const std::string& assetAbsPath) {
231 if (guid.empty() || assetAbsPath.empty()) {
232 return true;
233 }
234
235 std::string thumbPath = Path(guid);
236 if (thumbPath.empty() || !LxFile::Exists(thumbPath)) {
237 return true; // Thumbnail doesn't exist
238 }
239
240 // Get asset last write time
241 if (!LxFile::Exists(assetAbsPath)) {
242 return false; // Asset doesn't exist, can't rebuild
243 }
244
245 try {
246 auto assetTime = std::filesystem::last_write_time(assetAbsPath);
247 auto thumbTime = std::filesystem::last_write_time(thumbPath);
248
249 // If asset is newer than thumbnail, needs rebuild
250 return assetTime > thumbTime;
251 } catch (const std::exception& e) {
252 Log::Error("Thumbs: Failed to check rebuild need: " + std::string(e.what()));
253 return true; // Assume needs rebuild on error
254 }
255}

281// Process pending thumbnail builds
282void ProcessPendingBuilds(size_t maxBuilds) {
283 if (maxBuilds == 0) {
284 return;
285 }
286
287 std::vector<BuildTask> tasksToProcess;
288
289 {
290 std::lock_guard<std::mutex> lock(s_BuildQueueMutex);
291 size_t count = std::min(maxBuilds, s_BuildQueue.size());
292 for (size_t i = 0; i < count; ++i) {
293 if (!s_BuildQueue.empty()) {
294 tasksToProcess.push_back(s_BuildQueue.front());
295 s_BuildQueue.pop();
296 }
297 }
298 }
299
300 // Process tasks in background
301 for (const auto& task : tasksToProcess) {
302 ThreadPool::Submit([task]() {
303 Build(task.guid, task.type, task.assetAbsPath);
304 });
305 }
306}
