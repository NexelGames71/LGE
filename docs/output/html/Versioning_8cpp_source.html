176// Helper: Load version history
177static bool LoadHistory(const std::string& assetGuid, VersionHistory& history) {
178 std::string historyPath = GetHistoryPath(assetGuid);
179
180 if (!LxFile::Exists(historyPath)) {
181 return false;
182 }
183
184 std::string jsonStr;
185 IOError err = LxFile::ReadAll(historyPath, jsonStr);
186 if (err != IOError::None) {
187 return false;
188 }
189
190 // Simple JSON parsing for history
191 history.assetGuid = assetGuid;
192
193 // Find versions array
194 size_t pos = jsonStr.find("\"versions\"");
195 if (pos != std::string::npos) {
196 size_t arrayStart = jsonStr.find('[', pos);
197 size_t arrayEnd = jsonStr.find(']', arrayStart);
198 if (arrayStart != std::string::npos && arrayEnd != std::string::npos) {
199 // Parse version numbers
200 size_t elemStart = arrayStart + 1;
201 while (elemStart < arrayEnd) {
202 size_t numStart = jsonStr.find_first_of("0123456789", elemStart);
203 if (numStart == std::string::npos || numStart >= arrayEnd) break;
204 size_t numEnd = jsonStr.find_first_not_of("0123456789", numStart);
205 int version = std::stoi(jsonStr.substr(numStart, numEnd - numStart));
206
207 AssetVersion ver;
208 if (ReadVersionMeta(assetGuid, version, ver)) {
209 history.versions.push_back(ver);
210 }
211
212 elemStart = numEnd;
213 }
214 }
215 }
216
217 // Find current version
218 pos = jsonStr.find("\"currentVersion\"");
219 if (pos != std::string::npos) {
220 pos = jsonStr.find(':', pos);
221 if (pos != std::string::npos) {
222 size_t start = jsonStr.find_first_of("0123456789", pos);
223 size_t end = jsonStr.find_first_not_of("0123456789", start);
224 history.currentVersion = std::stoi(jsonStr.substr(start, end - start));
225 }
226 }
227
228 // Find latest version
229 pos = jsonStr.find("\"latestVersion\"");
230 if (pos != std::string::npos) {
231 pos = jsonStr.find(':', pos);
232 if (pos != std::string::npos) {
233 size_t start = jsonStr.find_first_of("0123456789", pos);
234 size_t end = jsonStr.find_first_not_of("0123456789", start);
235 history.latestVersion = std::stoi(jsonStr.substr(start, end - start));
236 }
237 }
238
239 return true;
240}

293int CreateVersion(const std::string& assetGuid,
294 const std::string& author,
295 const std::string& comment) {
296 if (!s_IsInitialized) {
297 return 0;
298 }
299
300 // Find asset
301 const AssetRecord* record = AssetRegistry::FindByGUID(assetGuid);
302 if (!record) {
303 Log::Error("Versioning: Asset not found: " + assetGuid);
304 return 0;
305 }
306
307 // Check if asset file exists
308 if (!LxFile::Exists(record->absPath)) {
309 Log::Error("Versioning: Asset file does not exist: " + record->absPath);
310 return 0;
311 }
312
313 // Load or create history
314 VersionHistory history;
315 if (!LoadHistory(assetGuid, history)) {
316 history.assetGuid = assetGuid;
317 history.currentVersion = 0;
318 history.latestVersion = 0;
319 }
320
321 // Create new version
322 int newVersion = history.latestVersion + 1;
323
324 AssetVersion version;
325 version.version = newVersion;
326 version.assetGuid = assetGuid;
327 version.checksum = CalculateChecksum(record->absPath);
328 version.timestamp = GetCurrentTime();
329 version.author = author;
330 version.comment = comment;
331 version.fileSize = LxFile::GetFileSize(record->absPath);
332 version.changeType = history.versions.empty() ? "create" : "modify";
333
334 // Create version directory
335 std::string assetDir = LxPath::GetParent(GetVersionFilePath(assetGuid, newVersion));
336 if (!LxFile::Exists(assetDir)) {
337 LxFile::CreateDirectory(assetDir);
338 }
339
340 // Copy asset file to version storage
341 std::string versionPath = GetVersionFilePath(assetGuid, newVersion);
342 IOError err = LxFile::Copy(record->absPath, versionPath, true);
343 if (err != IOError::None) {
344 Log::Error("Versioning: Failed to copy asset to version storage");
345 return 0;
346 }
347
348 // Save version metadata
349 if (!WriteVersionMeta(assetGuid, newVersion, version)) {
350 Log::Error("Versioning: Failed to save version metadata");
351 return 0;
352 }
353
354 // Update history
355 history.versions.push_back(version);
356 history.currentVersion = newVersion;
357 history.latestVersion = newVersion;
358
359 if (!SaveHistory(history)) {
360 Log::Error("Versioning: Failed to save version history");
361 return 0;
362 }
363
364 // Update in-memory cache
365 s_VersionHistory[assetGuid] = history;
366
367 Log::Info("Versioning: Created version " + std::to_string(newVersion) +
368 " for asset: " + assetGuid);
369
370 return newVersion;
371}

472VersionDiff CompareVersions(const std::string& assetGuid, int version1, int version2) {
473 VersionDiff diff;
474
475 auto ver1 = GetVersion(assetGuid, version1);
476 auto ver2 = GetVersion(assetGuid, version2);
477
478 if (!ver1.has_value() || !ver2.has_value()) {
479 return diff;
480 }
481
482 diff.hasChanges = (ver1->checksum != ver2->checksum);
483 diff.sizeDiff = (ver1->fileSize > ver2->fileSize) ?
484 (ver1->fileSize - ver2->fileSize) : (ver2->fileSize - ver1->fileSize);
485
486 // Simple similarity calculation based on checksum
487 if (ver1->checksum == ver2->checksum) {
488 diff.similarity = 1.0f;
489 } else {
490 diff.similarity = 0.0f; // Could be enhanced with actual diff algorithm
491 }
492
493 return diff;
494}

577bool TagVersion(const std::string& assetGuid, int version, const std::string& tag) {
578 auto ver = GetVersion(assetGuid, version);
579 if (!ver.has_value()) {
580 return false;
581 }
582
583 // Add tag if not already present
584 if (std::find(ver->tags.begin(), ver->tags.end(), tag) == ver->tags.end()) {
585 ver->tags.push_back(tag);
586 WriteVersionMeta(assetGuid, version, ver.value());
587
588 // Update cache
589 auto history = GetHistory(assetGuid);
590 if (history.has_value()) {
591 for (auto& v : history->versions) {
592 if (v.version == version) {
593 v.tags = ver->tags;
594 break;
595 }
596 }
597 s_VersionHistory[assetGuid] = history.value();
598 }
599 }
600
601 return true;

619VersionStats GetStats() {
620 VersionStats stats;
621
622 // Count assets with versions
623 for (const auto& [guid, history] : s_VersionHistory) {
624 stats.assetsWithVersions++;
625 stats.totalVersions += history.versions.size();
626
627 for (const auto& ver : history.versions) {
628 stats.totalVersionSize += ver.fileSize;
629
630 if (stats.oldestVersion == 0 || ver.timestamp < stats.oldestVersion) {
631 stats.oldestVersion = ver.timestamp;

633 if (ver.timestamp > stats.newestVersion) {
634 stats.newestVersion = ver.timestamp;
635 }
636 }
637 }
