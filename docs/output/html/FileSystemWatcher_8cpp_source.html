78void FileSystemWatcher::WatchThread() {
79 const auto pollInterval = std::chrono::milliseconds(500); // Poll every 500ms
80
81 while (!m_ShouldStop && m_IsWatching) {
82 std::this_thread::sleep_for(pollInterval);
83
84 if (!m_IsWatching) {
85 break;
86 }
87
88 try {
89 // Check for deleted files
90 auto it = m_FileCache.begin();
91 while (it != m_FileCache.end()) {
92 if (!std::filesystem::exists(it->path)) {
93 // File was deleted
94 if (m_EventCallback) {
95 FileSystemEvent event(FileSystemEventType::Deleted, it->path);
96 m_EventCallback(event);
97 }
98 it = m_FileCache.erase(it);
99 } else {
100 ++it;
101 }
102 }
103
104 // Check for new/modified files
105 ProcessDirectory(m_WatchedDirectory);
106
107 } catch (const std::exception& e) {

113void FileSystemWatcher::ProcessDirectory(const std::string& path) {
114 try {
115 for (const auto& entry : std::filesystem::recursive_directory_iterator(path)) {
116 if (m_ShouldStop) {
117 break;
118 }
119
120 if (ShouldIgnorePath(entry.path().string())) {
121 continue;
122 }
123
124 std::string filePath = entry.path().string();
125 uint64_t modTime = GetFileModificationTime(filePath);
126
127 // Check if file is already in cache
128 auto it = std::find_if(m_FileCache.begin(), m_FileCache.end(),
129 [&filePath](const FileInfo& info) {
130 return info.path == filePath;
131 });
132
133 if (it == m_FileCache.end()) {
134 // New file
135 m_FileCache.push_back({filePath, modTime});
136 if (m_EventCallback) {
137 FileSystemEvent event(FileSystemEventType::Created, filePath);
138 m_EventCallback(event);
139 }
140 } else {
141 // Check if modified
142 if (it->lastModified != modTime) {
143 it->lastModified = modTime;
144 if (m_EventCallback) {
145 FileSystemEvent event(FileSystemEventType::Modified, filePath);
146 m_EventCallback(event);
147 }
148 }
149 }
150 }
151 } catch (const std::exception& e) {
152 Log::Error("FileSystemWatcher: Exception processing directory: " + std::string(e.what()));
153 }

156uint64_t FileSystemWatcher::GetFileModificationTime(const std::string& path) {
157 try {
