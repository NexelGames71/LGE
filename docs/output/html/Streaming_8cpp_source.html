36
37 // Check if file exists

59 }

62 m_Position = 0;

116bool AssetStream::ReadChunk(size_t offset, size_t size) {
117 if (!m_IsOpen) {
118 return false;
119 }
120
121 // Check if chunk is already loaded
122 size_t chunkIndex = offset / m_ChunkSize;
123 if (chunkIndex >= m_Chunks.size()) {
124 return false;
125 }
126
127 auto it = m_LoadedChunks.find(chunkIndex);
128 if (it != m_LoadedChunks.end() && it->second.isLoaded) {
129 return true; // Already loaded
130 }
131
132 // Load chunk
133 StreamChunk chunk;
134 chunk.offset = m_Chunks[chunkIndex].offset;
135 chunk.size = m_Chunks[chunkIndex].size;
136 chunk.data.resize(chunk.size);
137
138 std::ifstream file(m_FilePath, std::ios::binary);
139 if (!file.is_open()) {
140 Log::Error("Streaming: Failed to open file for reading: " + m_FilePath);
141 return false;
142 }
143
144 file.seekg(chunk.offset);
145 file.read(reinterpret_cast<char*>(chunk.data.data()), chunk.size);
146 size_t bytesRead = file.gcount();
147 file.close();
148
149 if (bytesRead != chunk.size) {
150 Log::Error("Streaming: Failed to read chunk completely");
151 return false;
152 }
153
154 chunk.isLoaded = true;
155 m_LoadedChunks[chunkIndex] = chunk;
156 m_Chunks[chunkIndex].isLoaded = true;
157
158 return true;
159}

227std::unique_ptr<AssetStream> OpenStream(const std::string& assetGuid,
228 const StreamOptions& options) {

233 // Check if stream already exists

314bool StreamAssetToCallback(const std::string& assetGuid,
315 std::function<void(const void* data, size_t size)> callback,
316 const StreamOptions& options) {
317 if (assetGuid.empty() || !callback) {
318 return false;
319 }
320
321 // Open stream
322 auto stream = OpenStream(assetGuid, options);
323 if (!stream) {
324 return false;
325 }
326
327 // Stream data in chunks
328 size_t chunkSize = options.chunkSize;
329 std::vector<uint8_t> buffer(chunkSize);
330
331 size_t totalRead = 0;
332 size_t fileSize = stream->GetSize();
333
334 while (totalRead < fileSize) {
335 size_t toRead = std::min(chunkSize, fileSize - totalRead);
336 size_t bytesRead = stream->Read(buffer.data(), toRead);
337
338 if (bytesRead == 0) {
339 break; // EOF
340 }
341
342 // Call callback with chunk
343 callback(buffer.data(), bytesRead);
344
345 totalRead += bytesRead;
346
347 // Call progress callback if provided
348 if (options.progressCallback) {
349 options.progressCallback(totalRead, fileSize);
350 }
351 }
352
353 return true;
354}
