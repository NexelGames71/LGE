13// Check if a path contains a cyclic symlink
14bool CheckCyclicSymlink(const std::string& absPath) {
15 if (absPath.empty()) {
16 return false;
17 }
18
19 try {
20 std::filesystem::path path(absPath);
21 std::unordered_set<std::string> visited;
22
23 // Resolve symlinks and check for cycles
24 while (std::filesystem::exists(path) && std::filesystem::is_symlink(path)) {
25 std::string canonical = std::filesystem::canonical(path).string();
26
27 // Normalize for comparison
28 canonical = LxPath::Normalize(canonical);
29
30 // Check if we've seen this path before (cycle detected)
31 if (visited.find(canonical) != visited.end()) {
32 Log::Error("Policies: Cyclic symlink detected: " + absPath);
33 return true;
34 }
35
36 visited.insert(canonical);
37
38 // Follow symlink
39 path = std::filesystem::read_symlink(path);
40
41 // If relative symlink, resolve relative to parent
42 if (path.is_relative()) {
43 path = std::filesystem::path(absPath).parent_path() / path;
44 }
45 }
46
47 return false;
48 } catch (const std::exception& e) {
49 Log::Error("Policies: Failed to check symlink: " + std::string(e.what()));
50 return false; // Assume no cycle on error
51 }
52}

54// Validate filename according to platform rules
55Violation ValidateFilename(std::string_view filename) {
56 if (filename.empty()) {
57 return Violation::EmptyFilename;
58 }
59
60 // Check for control characters
61 for (char c : filename) {
62 if (std::iscntrl(static_cast<unsigned char>(c))) {
63 return Violation::ControlCharacter;
64 }
65 }
66
67 // Check for trailing spaces or dots (Windows requirement)
68 if (!filename.empty()) {
69 char last = filename.back();
70 if (last == ' ' || last == '.') {
71 return Violation::TrailingSpace;
72 }
73 }
74
75 // Check for invalid characters: < > : " / \ | ? *
76 for (char c : filename) {
77 if (c == '<' || c == '>' || c == ':' || c == '"' ||
78 c == '/' || c == '\\' || c == '|' || c == '?' || c == '*') {
79 return Violation::InvalidFilename;
80 }
81 }

84 #ifdef _WIN32

89 };
90

111// Check if operation is allowed between mounts
112Violation CheckCrossMountOperation(const std::string& srcMount, const std::string& dstMount, const std::string& operation) {
113 // Same mount - always allowed
114 if (srcMount == dstMount) {
115 return Violation::None;
116 }
117
118 // Moving to Packages is never allowed from UI
119 if (dstMount == "Packages" && operation == "move") {
120 return Violation::CrossMountToPackages;
121 }
122
123 // Moving from Packages to Assets is not allowed (must copy)
124 if (srcMount == "Packages" && dstMount == "Assets" && operation == "move") {
125 return Violation::CrossMountMove;
126 }
127
128 // Copying from Packages to Assets is allowed (but flagged)
129 if (srcMount == "Packages" && dstMount == "Assets" && operation == "copy") {
130 // Allowed, but should be flagged as user content duplication
131 return Violation::None;
132 }
133
134 // Moving from Assets to Packages is not allowed
135 if (srcMount == "Assets" && dstMount == "Packages" && operation == "move") {
136 return Violation::CrossMountToPackages;
137 }
138
139 // Other cross-mount operations are allowed
140 return Violation::None;
141}

143// Get mount name for an absolute path
144std::string GetMountName(const std::string& absPath) {
145 if (absPath.empty()) {
146 return "";
147 }
148
149 std::string normalized = LxPath::Normalize(absPath);
150
151 // Check each mount
152 auto mounts = VFS::ListMounts();
153 for (const auto& mount : mounts) {
154 if (LxPath::IsSubPath(normalized, mount.root)) {
155 return mount.name;
156 }
157 }
158
159 return "";
160}
