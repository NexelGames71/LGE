51static std::string NormalizePath(const std::string& path) {
52 std::string normalized = path;
53 std::replace(normalized.begin(), normalized.end(), '\\', '/');
54 // Remove trailing slash (except for root paths)
55 if (normalized.length() > 1 && normalized.back() == '/') {
56 normalized.pop_back();
57 }
