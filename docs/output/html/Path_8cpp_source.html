12// Normalize a path
13std::string Normalize(std::string_view path) {
14 if (path.empty()) {
15 return "";
16 }
17
18 try {
19 // Convert string_view to string
20 std::string pathStr(path);
21
22 // Handle Windows long path prefix (\\?\‍)
23 bool hasLongPathPrefix = false;
24 if (pathStr.length() >= 4 && pathStr.substr(0, 4) == "\\\\?\\") {
25 hasLongPathPrefix = true;
26 pathStr = pathStr.substr(4);
27 }
28
29 // Use std::filesystem for proper path normalization
30 std::filesystem::path fsPath(pathStr);
31
32 // Normalize the path
33 fsPath = fsPath.lexically_normal();
34
35 // Convert to string and normalize separators
36 std::string result = fsPath.string();
37 std::replace(result.begin(), result.end(), '\\', '/');
38
39 // Remove duplicate slashes
40 std::string normalized;
41 normalized.reserve(result.length());
42 bool lastWasSlash = false;
43 for (size_t i = 0; i < result.length(); ++i) {
44 if (result[i] == '/') {
45 if (!lastWasSlash || (i == 0 && result.length() > 1 && result[1] == '/')) {
46 // Keep first slash for UNC paths (//server/share)
47 normalized += '/';
48 }
49 lastWasSlash = true;
50 } else {
51 normalized += result[i];
52 lastWasSlash = false;
53 }
54 }
55
56 // Remove trailing slash (except for root paths)
57 if (normalized.length() > 1 && normalized.back() == '/') {
58 normalized.pop_back();
59 }
60
61 // Restore long path prefix if needed
62 if (hasLongPathPrefix) {
63 normalized = "\\\\?\\" + normalized;
64 }
65
66 return normalized;
67 } catch (const std::exception& e) {
68 Log::Error("LxPath: Failed to normalize path: " + std::string(e.what()));
69 // Fallback: simple normalization
70 std::string result(path);
71 std::replace(result.begin(), result.end(), '\\', '/');
72 return result;
73 }
74}

76// Join two path components
77std::string Join(std::string_view a, std::string_view b) {
78 if (a.empty()) {
79 return Normalize(b);
80 }
81 if (b.empty()) {
82 return Normalize(a);
83 }
84
85 std::string aNorm = Normalize(a);
86 std::string bNorm = Normalize(b);
87
88 // If b is absolute, return b
89 if (bNorm.length() > 0 && (bNorm[0] == '/' || (bNorm.length() > 1 && bNorm[1] == ':'))) {
90 return bNorm;
91 }
92
93 // Join with single slash
94 if (aNorm.back() == '/') {
95 return Normalize(aNorm + bNorm);
96 } else {
97 return Normalize(aNorm + "/" + bNorm);
98 }
99}

101// Check if child is a subpath of parent
102bool IsSubPath(std::string_view child, std::string_view parent) {
103 std::string childNorm = Normalize(child);
104 std::string parentNorm = Normalize(parent);
105
106 // Ensure parent ends with / for comparison
107 if (parentNorm.back() != '/') {
108 parentNorm += '/';
109 }
110
111 // Check if child starts with parent
112 if (childNorm.length() < parentNorm.length()) {
113 return false;
114 }
115
116 // Case-insensitive comparison on Windows, case-sensitive on Unix
117#ifdef _WIN32
118 std::string childLower = childNorm;
119 std::string parentLower = parentNorm;
120 std::transform(childLower.begin(), childLower.end(), childLower.begin(), ::tolower);
121 std::transform(parentLower.begin(), parentLower.end(), parentLower.begin(), ::tolower);
122 return childLower.substr(0, parentLower.length()) == parentLower;
123#else
124 return childNorm.substr(0, parentNorm.length()) == parentNorm;
125#endif
126}

192// Sanitize filename
193std::string SanitizeFilename(std::string_view filename) {
194 std::string result;
195 result.reserve(filename.length());
196
197 // Windows reserved names (case-insensitive)
198 static const char* reservedNames[] = {
199 "CON", "PRN", "AUX", "NUL",
200 "COM1", "COM2", "COM3", "COM4", "COM5", "COM6", "COM7", "COM8", "COM9",
201 "LPT1", "LPT2", "LPT3", "LPT4", "LPT5", "LPT6", "LPT7", "LPT8", "LPT9"
202 };
203
204 std::string upperName = std::string(filename);
205 std::transform(upperName.begin(), upperName.end(), upperName.begin(), ::toupper);
206
207 // Check for reserved names (with or without extension)
208 std::string stem = upperName;
209 size_t dotPos = stem.find('.');
210 if (dotPos != std::string::npos) {
211 stem = stem.substr(0, dotPos);
212 }
213
214 for (const char* reserved : reservedNames) {
215 if (stem == reserved) {
216 // Prepend underscore to make it valid
217 result += '_';
218 break;
219 }
220 }
221
222 // Remove invalid characters
223 for (char c : filename) {
224 // Invalid characters: < > : " / \ | ? *
225 if (c == '<' || c == '>' || c == ':' || c == '"' ||
226 c == '/' || c == '\\' || c == '|' || c == '?' || c == '*') {
227 result += '_';
228 } else if (std::iscntrl(static_cast<unsigned char>(c))) {
229 // Control characters
230 result += '_';
231 } else {
232 result += c;
233 }
234 }
235
236 // Remove trailing spaces and dots (Windows requirement)
237 while (!result.empty() && (result.back() == ' ' || result.back() == '.')) {
238 result.pop_back();
239 }
240
241 // Ensure filename is not empty
242 if (result.empty()) {
243 result = "_";
244 }
245
246 // Limit length (Windows MAX_PATH is 260, but we'll be more conservative)
247 if (result.length() > 255) {
248 result = result.substr(0, 255);
249 }
250
251 return result;
252}
